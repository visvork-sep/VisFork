21     export const lemmatizationFunction = (token: string) => {
22         if (typeof token !== "string") {
23             return token;
24         }
25     
26         // List of known prototype properties to exclude
27         const prototypeProperties = [
28             "constructor", "__proto__", "hasownproperty", 
29             "isprototypeof", "propertyisenumerable", 
30             "tolocalestring", "tostring", "valueof"
31         ];
32     
33         // Check if the token is a prototype property, which would cause an internal error
34         if (prototypeProperties.includes(token)) {
35             return token; // Return token as-is (could be changed for something better)
36         }
37     
38         try {
39             const lemmatized = lemmatizer(token);
40             if (typeof lemmatized !== "string") {
41                 console.error(`Lemmatizer returned non-string value for token "${token}":`, lemmatized);
42                 return token;
43             }
44             return lemmatized;
45         } catch (error) {
46             console.error(`Error in lemmatizer for token "${token}":`, error);
47             return token; // Fallback to the original token
48         }
49     };

52     export const processCommitMessages = (
53         data: string [],
54         processToken: (token: string) => string,
55         start: number,
56         finish: number,
57         maxSize = 100,
58         minSize = 25
59     ): Word[] => {
60         const wordFreq: Record<string, number> = {};
61     
62         data.forEach( (message: string ) => {
63             const tokens = tokenize(message);
64             const filteredTokens = removeStopwords(tokens).filter(token => /^[a-zA-Z]+$/.test(token));
65     
66             filteredTokens.forEach((token: string) => {
67                 let processedWord;
68                 try {
69                     processedWord = processToken(token.toLowerCase() + "");
70                     if (typeof processedWord !== "string" || processedWord.trim() === "") {
71                         throw new Error("Invalid processed word");
72                     }
73                 } catch (error) {
74                     console.error(`Error processing token "${token}":`, error);
75                     processedWord = token.toLowerCase(); // Fallback to original token
76                 }
77     
78                 wordFreq[processedWord] = (wordFreq[processedWord] || 0) + 1;
79             });
80         });
81     
82         // Convert frequencies to an array
83         let sortedWords = Object.entries(wordFreq)
84             .map(([word, freq]) => ({ text: word, frequency: freq, size: freq }))
85             .sort((a, b) => b.frequency - a.frequency)
86             .slice(Math.max(0, start), Math.min(Object.keys(wordFreq).length, finish));
87     
88         if (sortedWords.length === 0) return [];
89     
90         // Normalize sizes using log scaling
91         const maxFreq = sortedWords[0].frequency;
92     
93         sortedWords = sortedWords.map(({ text, frequency }) => ({
94             text,
95             frequency,
96             size: minSize + ((Math.log(frequency + 1) / Math.log(maxFreq + 1)) * (maxSize - minSize))
97         }));
98     
99     
100         return sortedWords;
101     };
