1     import { useRef, useEffect, useState, useMemo, useCallback, memo } from "react";
2     import { select } from "d3-selection";
3     import { brush } from "d3-brush";
4     import type { D3BrushEvent } from "d3-brush";
5     import {
6         graphStratify,
7         grid,
8         tweakFlip,
9         laneGreedy,
10         MutGraph,
11         MutGraphNode,
12     } from "d3-dag";
13     
14     import { TimelineProps, TimelineDetails as Commit } from "@VisInterfaces/TimelineData";
15     import { GroupedNode, NodeSelection } from "./timelineUtils";
16     import * as utils from "./timelineUtils";
17     import * as graphics from "./timelineGraphics";
18     import * as c from "./timelineConstants";
19     import { themeGet, useTheme } from "@primer/react";
20     
21     function CommitTimeline({
22         commitData,
23         handleTimelineSelection,
24     }: TimelineProps) {
25         const [merged, setMerged] = useState(false); // State for merged view
26         const [selectAll, setSelectAll] = useState(false); // State for selection
27     
28         const svgRef = useRef<SVGSVGElement | null>(null);
29         const containerRef = useRef<HTMLDivElement>(null); // Parent container
30     
31         // Color controls
32         const { colorMode, theme } = useTheme();
33         const isDarkMode = useMemo(() => {
34             return colorMode === "dark";
35         }, [colorMode]);
36         const colorScheme = useMemo(() => {
37             return { 
38                 neutralLaneColor : themeGet("colors.neutral.muted")({ theme }),
39                 accentedLaneColor : themeGet("colors.accent.muted")({ theme }),
40                 markerColor : themeGet("colors.fg.subtle")({ theme }),
41                 overlayColor : themeGet("colors.fg.muted")({ theme })};
42         }, [theme]);
43     
44         // Buttons style and hover/leave design
45         const BUTTON_STYLE: React.CSSProperties = {
46             width: "120px", height: "40px", padding: "5px 10px",
47             color: `${isDarkMode ? "white" : "black"}`,
48             cursor: "pointer", backgroundColor: "transparent",
49             border: `2px solid ${colorScheme.markerColor}`, display: "flex",
50             alignItems: "center", justifyContent: "center",
51             transition: "background-color 0.3s",
52         };
53     
54         const onHover = (e: React.MouseEvent<HTMLButtonElement>) => 
55             e.currentTarget.style.backgroundColor = colorScheme.accentedLaneColor;
56         const onLeave = (e: React.MouseEvent<HTMLButtonElement>) => 
57             e.currentTarget.style.backgroundColor = "transparent";
58         
59         // Memoize the color map
60         const colorMap = useMemo(() => utils.buildRepoColorMap(commitData), [commitData]);
61     
62         // Precompute both views, update only when data changes
63         const builder = graphStratify();
64         const dagMerged = useMemo(() => {
65             try {
66                 return builder(utils.groupNodes(commitData) as GroupedNode[]);
67             } catch (error) {
68                 console.error("Error building merged DAG:", error);
69                 return null;
70             }
71         }, [commitData]);
72     
73         const dagFull = useMemo(() => {
74             try {
75                 return builder(commitData as Commit[]);
76             } catch (error) {
77                 console.error("Error building full DAG:", error);
78                 return null;
79             }
80         }, [commitData]);
81     
82         // Draw the graph
83         const drawGraph = useCallback(() => {
84             if (!svgRef.current || !commitData.length || !dagMerged || !dagFull) return;
85     
86             // Clear previous visualization
87             select(svgRef.current).selectAll("*").remove();
88             select("#dag-legends").selectAll("*").remove();
89     
90             const dag = merged
91                 ? (dagMerged as MutGraph<Commit | GroupedNode, undefined>)
92                 : (dagFull as MutGraph<Commit | GroupedNode, undefined>);
93     
94             const layout = grid()
95                 .nodeSize([c.NODE_RADIUS * 2, c.NODE_RADIUS * 2])
96                 .gap([5, 5])
97                 .tweaks([tweakFlip("diagonal")])
98                 .rank(utils.dateRankOperator)
99                 .lane(laneGreedy().topDown(true).compressed(false));
100     
101             // Initial dimensions, will be overwritten
102             let { width, height } = layout(dag);
103     
104             width = Math.max(
105                 width + c.MARGIN.left + c.MARGIN.right,
106                 containerRef.current?.clientWidth ?? 0
107             );
108     
109             const svg = select(svgRef.current)
110                 .attr("width", width)
111                 .attr("height", height + c.MARGIN.top + c.MARGIN.bottom)
112                 .attr("viewBox", `0 0 ${width} ${height + c.MARGIN.top + c.MARGIN.bottom}`);
113             const g = svg
114                 .append("g")
115                 .attr("transform", `translate(${c.MARGIN.left},${c.MARGIN.top})`);
116     
117             // Sort nodes by date
118             const sortedNodes =
119                 Array.from(dag.nodes()).sort((a, b) => {
120                     return new Date(a.data.date).getTime() - new Date(b.data.date).getTime();
121                 });
122     
123             // Apply custom layout
124             const { lanes, totalHeight } = utils.assignUniqueLanes(sortedNodes);
125             height = totalHeight + c.MARGIN.top + c.MARGIN.bottom;
126             // Adjust height to custom layout
127             select(svgRef.current)
128                 .attr("height", height)
129                 .attr("viewBox", `0 0 ${width} ${height}`);
130     
131             // Lane shading and author labels
132             graphics.drawLanes(g, lanes, width, isDarkMode, colorScheme.neutralLaneColor, colorScheme.accentedLaneColor);
133     
134             // month/year labels
135             graphics.drawTimelineMarkers(g, sortedNodes, totalHeight, isDarkMode, colorScheme.markerColor, merged);
136     
137             // Selection overlay for selectAll
138             if (selectAll) {
139                 g.append("rect")
140                     .attr("class", "selection-overlay")
141                     .attr("x", 0)
142                     .attr("y", c.MARGIN.top)
143                     .attr("width", width)
144                     .attr("height", totalHeight - c.MARGIN.bottom - c.MARGIN.top)
145                     .style("fill", colorScheme.overlayColor)
146                     .style("fill-opacity", "0.3");
147             } else {
148                 g.select(".selection-overlay").remove();
149             }
150     
151             let brushSelection: [[number, number], [number, number]] = [[0, 0], [0, 0]];
152     
153             const brushStart = (event: D3BrushEvent<unknown>) => {
154                 setSelectAll(false);
155                 if (event.sourceEvent && event.sourceEvent.type !== "end") {
156                     brushSelection = [[0, 0], [0, 0]];
157                 }
158             };
159     
160             const brushEnd =
161                 (event: D3BrushEvent<unknown>) => {
162                     if (event.selection === null || !event.sourceEvent) return; // Exit if no selection
163     
164                     brushSelection = event.selection as [[number, number], [number, number]];
165                     const [x0, y0] = brushSelection[0];
166                     const [x1, y1] = brushSelection[1];
167     
168                     const selectedCommits = sortedNodes
169                         .filter((node) => {
170                             const x = node.x + c.NODE_RADIUS;
171                             const y = node.y + c.NODE_RADIUS;
172                             return x >= x0 && x <= x1 && y >= y0 && y <= y1;
173                         })
174                         .flatMap((node) =>
175                             merged
176                                 ? (node as MutGraphNode<GroupedNode, unknown>).data.nodes
177                                 : [node.data.id]
178                         );
179     
180                     handleTimelineSelection(selectedCommits);
181                 };
182     
183             const timelineBrush = brush<unknown>()
184                 .extent([
185                     [-c.MARGIN.left, 0],
186                     [width - c.MARGIN.left, totalHeight],
187                 ])
188                 .on("start", brushStart)
189                 .on("end", brushEnd);
190     
191             // Create edges
192             g.append("g")
193                 .selectAll("path")
194                 .data(dag.links())
195                 .enter()
196                 .append("path")
197                 .attr("fill", "none")
198                 .attr("stroke", colorScheme.markerColor)
199                 .attr("stroke-width", c.EDGE_WIDTH)
200                 .attr("d", graphics.drawEdgeCurve);
201     
202             const brushLayer = g.append("g")
203                 .attr("class", "brush-layer");
204                 
205             // Apply brush to this layer only (otherwise conflicts with node clicking)
206             brushLayer.call(timelineBrush);
207     
208             const resetBrushing = () => {
209                 brushLayer.call(timelineBrush.move, [[0,0], [0,0]]);
210             };
211     
212             // Create tooltip
213             const tooltip = select("body")
214                 .append("div")
215                 .attr("class", "tooltip")
216                 .style("position", "absolute")
217                 .style("background", c.TOOLTIP_BG_COLOR)
218                 .style("padding", c.TOOLTIP_PADDING)
219                 .style("border", c.TOOLTIP_BORDER)
220                 .style("border-radius", c.TOOLTIP_BORDER_RADIUS)
221                 .style("pointer-events", "none")
222                 .style("opacity", 0)
223                 .style("font", c.TOOLTIP_FONT);
224     
225             // Draw nodes depending on view
226             if (merged) {
227                 const mergedNodes = sortedNodes as unknown as MutGraphNode<GroupedNode, undefined>[];
228                 const { circles, squares, triangles } = graphics.drawMergedNodes(g, colorMap, mergedNodes);
229     
230                 applyToolTip(circles as NodeSelection);
231                 applyToolTip(triangles as NodeSelection);
232                 applyToolTip(squares as unknown as NodeSelection);
233     
234             } else {
235                 const { circles } = graphics.drawNormalNodes(g, colorMap, sortedNodes);
236     
237                 // Apply tooltips
238                 applyToolTip(circles as NodeSelection);
239             }
240     
241             function applyToolTip(selection: NodeSelection) {
242                 selection
243                     .on("mouseover", (event, d) => {
244                         tooltip.transition().duration(c.TOOLTIP_MOUSEOVER_DUR).style("opacity", 0.9);
245                         let content = "";
246                         if (!merged) {
247                             // Full view
248                             const commitData = d.data as Commit;
249                             content = `
250                             <strong>Commit</strong>: ${commitData.id}<br>
251                             <strong>Repo</strong>: ${commitData.repo}<br>
252                             <strong>Branch</strong>: ${commitData.branch}<br>
253                             <strong>Date</strong>: ${commitData.date}`;
254                         } else {
255                             // Merged view 
256                             const mergedData = d.data as GroupedNode;
257                             content = `
258                             <strong>Type of Commit</strong>: ${mergedData.branch}<br>
259                             <strong>Repo</strong>: ${mergedData.repo}<br>
260                              ${mergedData.nodes.length > 1
261             ? `<strong>Number of Commits</strong>: ${mergedData.nodes.length}<br>
262                                 <strong>Date of First Commit</strong>: ${mergedData.date}<br>
263                                 <strong>Date of Last Commit</strong>: ${mergedData.end_date}`
264             : `<strong>Date</strong>: ${mergedData.date}`}`;
265                         }
266                         tooltip.html(content)
267                             .style("left", (event.pageX) + "px")
268                             .style("top", (event.pageY) + "px");
269                     })
270                     .on("mousemove", (event) => {
271                         tooltip.style("left", (event.pageX) + "px")
272                             .style("top", (event.pageY) + "px");
273                     })
274                     .on("mouseout", () => {
275                         tooltip.transition().duration(c.TOOLTIP_MOUSEOUT_DUR).style("opacity", 0);
276                     })
277                     .on("click", (_event, d) => {
278                         _event.stopPropagation();
279                         if (!merged || d.data.branch != "default") {
280                             window.open(d.data.url);
281                         }
282                     });
283             }
284     
285             // Display legends for the colors in #dag-legends
286             const legend = select("#dag-legends")
287                 .style("display", "flex")
288                 .style("align-items", "flex-start");
289     
290             graphics.drawLegends(
291                 merged, legend, colorMap, 
292                 colorScheme.markerColor, sortedNodes, 
293                 setSelectAll, handleTimelineSelection, resetBrushing);
294     
295             return () => {
296                 tooltip.remove(); // Remove tooltip on cleanup
297             };
298         }, [commitData, merged, selectAll, theme]);
299     
300         useEffect(() => {
301             drawGraph();
302             window.addEventListener("resize", drawGraph);
303             return () => window.removeEventListener("resize", drawGraph);
304         }, [drawGraph]);
305     
306         return (
307             <div ref={containerRef} style={{ width: "100%", height: "100%" }}>
308                 {/* Button at the top */}
309                 <div
310                     style={{
311                         padding: "10px",
312                         background: "transparent",
313                         borderBottom: `1px solid ${colorScheme.markerColor}`,
314                         textAlign: "left",
315                     }}
316                 >
317                     <div style={{ display: "flex", gap: "10px" }}>
318                         <button
319                             onClick={() => setMerged(!merged)}
320                             style={BUTTON_STYLE}
321                             onMouseEnter={onHover}
322                             onMouseLeave={onLeave}
323                         >
324                             {merged ? "Full View" : "Merged View"}
325                         </button>
326     
327                         <button
328                             onClick={() => {
329                                 const selectedCommits = !selectAll
330                                     ? commitData.map((n) => n.id) // Select all
331                                     : []; // Deselect all
332                                 handleTimelineSelection(!selectAll ? selectedCommits : []);
333                                 setSelectAll(!selectAll);
334                             }}
335                             style={BUTTON_STYLE}
336                             onMouseEnter={onHover}
337                             onMouseLeave={onLeave}
338                         >
339                             {selectAll ? "Deselect All" : "Select All"}
340                         </button>
341                     </div>
342                 </div>
343     
344                 {/* Scrollable container for the SVG */}
345                 <div
346                     style={{
347                         overflow: "auto",
348                     }}
349                 >
350                     <svg ref={svgRef}></svg>
351                 </div>
352     
353                 {/* Legends at the bottom */}
354                 <div
355                     id="dag-legends"
356                     style={{
357                         padding: "10px",
358                         background: "transparent",
359                         borderTop: `1px solid ${colorScheme.markerColor}`,
360                     }}
361                 ></div>
362             </div>
363         );
364     };
365     
366     export default memo(CommitTimeline);
