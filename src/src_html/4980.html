5     /**
6          * Determines whether {@param fork} is valid or not.
7          * 
8          * @returns True if {@code fork} is valid according to the {@param filter}, false otherwise.
9          * 
10          */
11     function isValidForkByFilter(fork: UnprocessedRepository, filter: ForkFilter): boolean {
12     
13         if (!isForkWithOwnerOfType(fork, filter.ownerTypes)) {
14             return false;
15         }
16     
17         if (fork.last_pushed &&
18             filter.updatedInLastMonths &&
19             !isForkUpdatedInLastMonths(fork, filter.updatedInLastMonths)
20         ) {
21             return false;
22         }
23     
24         return true;
25     }

27     /**
28      * If {@param ownerTypes} is not undefined or null, this function determines whether
29      * the owner of {@param fork} is of any of the types in {@param ownerTypes}.
30      * 
31      * @returns True if {@code onwerTypes.includes(fork.ownerType).
32      */
33     function isForkWithOwnerOfType(fork: UnprocessedRepository, ownerTypes: OwnerType[]): boolean {
34         return ownerTypes.includes(fork.ownerType);
35     }

37     /**
38      * If {@param nrOfMonths} is not undefined or null, this function determines whether
39      * the {@param fork} was updated in the previous {@code nrOfMonths}.
40      * 
41      * If {@code fork.last_pushed} is undefined or null, the function will fail, 
42      * Typecheck before using
43      * 
44      * @throws TypeError, if {@param fork.last_pushed} is null or undefined.
45      * 
46      * @returns True if {@code fork} was updated in the previous {@code nrOfMonths}, false otherwise.
47      */
48     function isForkUpdatedInLastMonths(fork: UnprocessedRepository, nrOfMonths: number): boolean {
49         if (!fork.last_pushed) {
50             throw TypeError("Fork last pushed date is null or undefined");
51         }
52         const thresholdDate = new Date();
53         thresholdDate.setMonth(new Date().getMonth() - nrOfMonths);
54     
55         return fork.last_pushed >= thresholdDate;
56     }
