1     import { EyeClosedIcon, EyeIcon, MoonIcon, SunIcon } from "@primer/octicons-react";
2     import { ActionMenu, Avatar, Box, Button, Dialog, Stack, useTheme } from "@primer/react";

5     import { useAuth } from "@Providers/AuthProvider";

7     import { AUTH_URL } from "@Utils/Constants";

11     /**
12      * Redirects the user to GitHub OAuth login.
13      *
14      * This function initiates the authentication flow by redirecting the user
15      * to the GitHub OAuth authorization endpoint. A small timeout is used
16      * to accommodate Safari's behavior with redirects.
17      */
18     function redirectLogin() {
19         setTimeout(() => { // timeout for Safari behavior
20             window.location.href = AUTH_URL + "/auth/github";
21         }, 250);
22     }
23     
24     function getLoginOrAvatar(isAuthenticated: boolean,
25         onDialogOpen: () => void,
26         avatarUrl: string | undefined,
27     ) {
28         return isAuthenticated ?
29             (
30                 <Box onClick={onDialogOpen}>
31                     {avatarUrl ?
32                         <Avatar src={avatarUrl} size={32} alt="user avatar" data-testid="user-avatar" />
33     
34                         :
35                         <SkeletonAvatar size={32} />
36                     }
37                 </Box>
38             )
39             :
40             (
41                 <Button onClick={redirectLogin}>Sign in</Button>
42             );
43     }

45     /**
46      * AppHeader Component
47      *
48      * This component represents the navigation bar for the application, handling user authentication,
49      * display settings (dark mode, colorblind mode), and a side panel for additional user options.
50      */
51     function AppHeader() {
52         /** Controls the visibility of the settings side panel */
53         const { isAuthenticated, logout } = useAuth();
54         const { data } = useFetchAvatarUrl({});
55         const avatarUrl = data?.viewer.avatarUrl as string | undefined;
56     
57         const { colorMode, setColorMode, dayScheme, nightScheme, setDayScheme, setNightScheme } = useTheme();
58         const [isOpen, setIsOpen] = useState(false);
59     
60         const onDialogClose = useCallback(() => setIsOpen(false), []);
61         const onDialogOpen = useCallback(() => setIsOpen(true), []);
62     
63         const handleLogout = useCallback(() => {
64             logout();
65             setIsOpen(false);
66         }, []);
67     
68         const loginOrAvatar = getLoginOrAvatar(isAuthenticated, onDialogOpen, avatarUrl);
69     
70         const currentlyColorblindMode = dayScheme === "light_colorblind" || nightScheme === "dark_colorblind";
71         const onToggleColorblindMode = useCallback(() => {
72             const newDayScheme = currentlyColorblindMode ? "light" : "light_colorblind";
73             const newNightScheme = currentlyColorblindMode ? "dark" : "dark_colorblind";
74             setDayScheme(newDayScheme);
75             setNightScheme(newNightScheme);
76         }, [currentlyColorblindMode, setDayScheme, setNightScheme]);
77     
78         const currentlyDarkMode = colorMode === "dark" || colorMode === "night";
79         const onToggleDarkMode = useCallback(() => {
80             const newMode = currentlyDarkMode ? "day" : "dark";
81             setColorMode(newMode);
82         }, [currentlyDarkMode, setColorMode]);
83         return (
84             <>
85                 <Stack direction="horizontal" align="center">
86                     <Stack.Item>
87                         <Avatar src={visForkIcon} size={32} alt="logo avatar" data-testid="logo-avatar" />
88                     </Stack.Item>
89                     <Stack.Item grow>
90                         <span>VisFork</span>
91                     </Stack.Item>
92                     <Stack.Item>
93                         {loginOrAvatar}
94                     </Stack.Item>
95                 </Stack>
96     
97                 {/* Side panel for user settings, opens when 'isOpen' is true */}
98                 {isOpen && (
99                     <Dialog title="Settings" onClose={onDialogClose} position="right" width="small">
100                         <ActionMenu>
101                             <ActionList>
102                                 <ActionList.Item
103                                     onClick={onToggleColorblindMode}
104                                     aria-label="Toggle colorblind mode"
105                                 >
106                                     <ActionList.LeadingVisual>
107                                         {currentlyColorblindMode ? <EyeClosedIcon /> : <EyeIcon />}
108                                     </ActionList.LeadingVisual>
109                                     {currentlyColorblindMode ? "Colorblind mode off" : "Colorblind mode on"}
110                                 </ActionList.Item>
111     
112                                 <ActionList.Item
113                                     onClick={onToggleDarkMode}
114                                     aria-label="Toggle light mode"
115                                 >
116                                     <ActionList.LeadingVisual>
117                                         {currentlyDarkMode ? <SunIcon /> : <MoonIcon />}
118                                     </ActionList.LeadingVisual>
119                                     {currentlyDarkMode ? "Light mode" : "Dark mode"}
120                                 </ActionList.Item>
121     
122                                 <ActionList.Divider />
123     
124                                 <ActionList.Item onClick={handleLogout} variant="danger" aria-label="Sign out button">
125                                     Sign out
126                                 </ActionList.Item>
127                             </ActionList>
128                         </ActionMenu>
129                     </Dialog>
130                 )}
131             </>
132         );
133     }
134     export default AppHeader;
