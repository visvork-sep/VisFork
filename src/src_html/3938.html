14     const mapCommitDataToHistogram = (commitData: Commit[]): HistogramData => ({
15         commitData: commitData.map((commit) => ({
16             repo: commit.repo,
17             date: commit.date,
18         })),
19     });

21     const mapCommitDataToTimeline = (commitData: Commit[]): TimelineData => ({
22         commitData: commitData.map((commit) => ({
23             repo: commit.repo,
24             id: commit.sha,
25             parentIds: commit.parentIds,
26             branch: commit.branch,
27             date: commit.date.toISOString(),
28             url: commit.url,
29         })),
30     });
31     
32     const mapCommitDataToCommitTable = (commitData: Commit[]): CommitTableData => ({
33         commitData: commitData.map((commit) => ({
34             id: commit.sha,
35             repo: commit.repo,
36             author: commit.author,
37             login: commit.login,
38             date: commit.date.toISOString(),
39             message: commit.message,
40         })),
41     });

43     const mapCommitDataToWordCloud = (commitData: Commit[]): WordCloudData => ({

63     export function useVisualizationData(forkData: Repository[], commitData: Commit[]) {
64         // Memoize the initial visualization data
65         const initialVisData = useMemo(() => {
66             const forkListData: ForkListData = { forkData: forkData };
67     
68             return {
69                 forkListData,
70                 histogramData: mapCommitDataToHistogram(commitData),
71                 timelineData: mapCommitDataToTimeline(commitData),
72                 commitTableData: mapCommitDataToCommitTable(commitData),
73                 wordCloudData: mapCommitDataToWordCloud(commitData),
74                 sankeyData: mapCommitDataToSankey(commitData),
75                 collabGraphData: mapCommitDataToCollabGraph(commitData),
76             };
77         }, [forkData, commitData]);
78     
79         const [visData, setVisData] = useState<VisualizationData>(initialVisData);
80     
81         useEffect(() => setVisData(initialVisData), [forkData, commitData]);
82     
83         // Handlers
84         const handleHistogramSelection = useCallback(
85             (startDate: Date, endDate: Date) => {
86                 // Filter commits based on date range
87                 const filteredCommits = commitData.filter(
88                     (commit) => new Date(commit.date) >= startDate && new Date(commit.date) <= endDate
89                 );
90     
91                 // Create a Set of valid commit IDs for fast lookup
92                 const validCommitIds = new Set(filteredCommits.map(commit => commit.sha));
93     
94                 // Remove parents that are outside the valid set
95                 const constrainedCommits = filteredCommits.map(commit => ({
96                     ...commit,
97                     parentIds: commit.parentIds.filter(parentId => validCommitIds.has(parentId))
98                 }));
99             
100                 setVisData((prev) => ({
101                     ...prev,
102                     timelineData: mapCommitDataToTimeline(constrainedCommits),
103                     sankeyData: mapCommitDataToSankey(constrainedCommits),
104                     collabGraphData: mapCommitDataToCollabGraph(constrainedCommits),
105                     commitTableData: mapCommitDataToCommitTable(constrainedCommits),
106                     wordCloudData: mapCommitDataToWordCloud(constrainedCommits),
107                 }));
108             },
109             [commitData]
110         );
111     
112         const handleTimelineSelection = useCallback(
113             (hashes: string[]) => {
114                 const constrainedCommits = commitData.filter((commit) =>
115                     hashes.includes(commit.sha)
116                 );
117     
118                 setVisData((prev) => ({
119                     ...prev,
120                     sankeyData: mapCommitDataToSankey(constrainedCommits),
121                     collabGraphData: mapCommitDataToCollabGraph(constrainedCommits),
122                     commitTableData: mapCommitDataToCommitTable(constrainedCommits),
123                     wordCloudData: mapCommitDataToWordCloud(constrainedCommits),
124                 }));
125             },
126             [commitData]
127         );
128     
129         const handleSearchBarSubmission = useCallback(
130             (hashes: string[]) => {
131                 const constrainedCommits = commitData.filter((commit) =>
132                     hashes.includes(commit.sha)
133                 );
134     
135                 setVisData((prev) => ({
136                     ...prev,
137                     wordCloudData: mapCommitDataToWordCloud(constrainedCommits),
138                     sankeyData: mapCommitDataToSankey(constrainedCommits),
139                     collabGraphData: mapCommitDataToCollabGraph(constrainedCommits),
140                 }));
141             },
142             [commitData]
143         );
144     
145         const defaultBranches = useMemo(() => {
146             const branches = forkData.reduce((acc, fork) => {
147                 if (!acc[fork.name]) {
148                     acc[fork.name] = fork.defaultBranch;
149                 }
150                 return acc;
151             }, {} as Record<string, string>);
152     
153             return branches;
154         }, [forkData]);
155     
156         return {
157             visData,
158             handlers: {
159                 handleHistogramSelection,
160                 handleTimelineSelection,
161                 handleSearchBarSubmission
162             },
163             defaultBranches
164         };
165     }
