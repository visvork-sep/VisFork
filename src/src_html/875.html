11     function DataComponents() {
12         const { onFiltersChange, forks, commits, filters, isLoading } = useFilteredData();
13     
14         // Main repo is the first member of the forks list
15         const mainRepoName = useMemo(() => {
16             return `${forks[0]?.owner.login}/${forks[0]?.name}`;
17         }, [forks]);
18     
19         const defaultBranchesMap = useMemo(() => createDefaultBranchesMap(forks), [forks]);
20         const filteredCommits = useMemo(() => processCommits(commits, defaultBranchesMap, mainRepoName), [
21             commits,
22             defaultBranchesMap,
23             mainRepoName,
24         ]);
25     
26         const applicationBody = useMemo(
27             () => {
28                 const { commits: processedCommits, forks: processedForks } = preprocessor(filteredCommits, forks, filters);
29                 return <SplitPageLayout.Content aria-label="Content" width="xlarge">
30                     <ApplicationBody forks={processedForks} commits={processedCommits} />
31                 </SplitPageLayout.Content>;
32             },
33             [filteredCommits, forks]
34         );
35     
36         const configurationPane = useMemo(() => {
37             return (
38                 <SplitPageLayout.Pane aria-label="Pane" width="medium" resizable>
39                     <ConfigurationPane filterChangeHandler={onFiltersChange} isDataLoading={isLoading} />
40                 </SplitPageLayout.Pane>
41             );
42         }, [isLoading]);
43     
44         return (
45             <>
46                 {configurationPane}
47                 {applicationBody}
48             </>
49         );
50     }

52     function preprocessor(commits: UnprocessedCommitExtended[],
53         forks: UnprocessedRepository[], filter?: ForkFilter): { forks: Repository[], commits: Commit[]; } {
54         const processedForks: Repository[] = forks.map(fork => ({
55             id: fork.id,
56             name: `${fork.owner.login}/${fork.name}`,
57             owner: { login: fork.owner.login },
58             description: fork.description ?? "",
59             created_at: fork.created_at ?? new Date(),
60             last_pushed: fork.last_pushed ?? new Date(),
61             ownerType: fork.ownerType,
62             defaultBranch: fork.defaultBranch,
63         }));
64     
65         const processedCommits: Commit[] = commits.map(commit => ({
66             sha: commit.sha,
67             id: commit.id,
68             parentIds: commit.parentIds,
69             node_id: commit.node_id,
70             mergedNodes: [],
71             date: commit.date === "Unknown" ? new Date() : commit.date,
72             url: commit.url,
73             author: commit.author,
74             login: commit.login,
75             message: commit.message,
76             commitType: classify(commit.message),
77             branch: commit.branch ?? "",
78             repo: commit.repo ?? "",
79         })).filter(commit => {
80             return filter ? filter.commitTypes.includes(commit.commitType) : true;
81         });
82     
83         return { forks: processedForks, commits: processedCommits };
84     }

86     function createDefaultBranchesMap(repos: UnprocessedRepository[]): Record<string, string> {
87         const defaultBranches: Record<string, string> = {};
88     
89         for (const repo of repos) {
90             const fullName = `${repo.owner.login}/${repo.name}`;
91             defaultBranches[fullName] = repo.defaultBranch;
92         }
93     
94         return defaultBranches;
95     }
