1     import { describe, it, expect, vi, afterEach } from "vitest";
2     import { render, screen, fireEvent, waitFor } from "@testing-library/react";
3     import { ThemeProvider } from "@primer/react";
4     import ForkList from "./ForkList";
5     import { ForkListData, ForkListDetails } from "@VisInterfaces/ForkListData";
6     
7     // Sample fork data 
8     const forkDetails: ForkListDetails[] = Array.from({ length: 25 }, (_, i) => ({
9         id: i + 1,
10         name: `fork/repo${i + 1}`,
11         description: `Description for fork ${i + 1}`,
12     }));
13     
14     const forkListData: ForkListData = {
15         forkData: forkDetails
16     };
17     
18     // quick helper to render the component 
19     const renderForkList = (
20         data = forkListData,
21     ) => {
22         return render(
23             <ThemeProvider>
24                 <ForkList forkData={data.forkData} />
25             </ThemeProvider>
26         );
27     };
28     
29     afterEach(() => {
30         vi.restoreAllMocks();
31         // clear any DOM changes
32         document.body.innerHTML = "";
33     });
34     
35     describe("ForkList tests", () => {
36         it("renders without errors when forkData is empty", () => {
37             renderForkList({ forkData: [] });
38             expect(screen.getByText("Forks")).toBeInTheDocument();
39             expect(screen.getByText("A list of all the forks of the submitted repository.")).toBeInTheDocument();
40             // Table should be empty
41             expect(screen.queryByRole("row", { name: /fork/i })).not.toBeInTheDocument();
42         });
43     
44         it("renders table with default page size (10)", () => {
45             renderForkList();
46             // Should render only 10 rows of data
47             for (let i = 1; i <= 10; i++) {
48                 expect(screen.getByText(`fork/repo${i}`)).toBeInTheDocument();
49             }
50             // 11th row should not be in DOM yet
51             expect(screen.queryByText("fork/repo11")).not.toBeInTheDocument();
52         });
53     
54         it("changes page size to 5 and updates table", async () => {
55             renderForkList();
56     
57             const select = screen.getByRole("combobox");
58             fireEvent.change(select, { target: { value: "5" } });
59     
60             await waitFor(() => {
61                 expect(screen.getByText("fork/repo1")).toBeInTheDocument();
62                 expect(screen.queryByText("fork/repo6")).not.toBeInTheDocument();
63             });
64         });
65     
66         it("navigates to next page", async () => {
67             renderForkList();
68     
69             // Navigate to page 2 (pageIndex = 1)
70             const nextButton = screen.getByRole("button", { name: /next/i });
71             fireEvent.click(nextButton);
72     
73             await waitFor(() => {
74                 // Should now be showing items 11â€“20
75                 expect(screen.getByText("fork/repo11")).toBeInTheDocument();
76                 expect(screen.queryByText("fork/repo1")).not.toBeInTheDocument();
77             });
78         });
