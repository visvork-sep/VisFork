15     describe("buildRepoColorMap", () => {
16         it("returns correct colors for unique repos", () => {
17             const commits = [
18                 { id: "1", repo: "A", date: "2021-01-01", parentIds: [], branch: "", url: "" },
19                 { id: "2", repo: "B", date: "2021-01-02", parentIds: [], branch: "", url: "" },
20                 { id: "3", repo: "A", date: "2021-01-03", parentIds: [], branch: "", url: "" },
21             ];
22             // The order of unique repos is preserved as first occurrence: ["A", "B"]
23             const uniqueRepos = Array.from(new Set(commits.map((c) => c.repo)));
24             const repoCount = uniqueRepos.length; // should be 2
25             const colorMap = buildRepoColorMap(commits);
26             expect(colorMap.size).toBe(repoCount);
27             // Verify that each repo gets the correct color based on its index.
28             expect(colorMap.get("A")).toBe(interpolateRainbow(0 / repoCount));
29             expect(colorMap.get("B")).toBe(interpolateRainbow(1 / repoCount));
30         });
31     });

63     describe("groupNodes()", () => {
64         it("should group commits into one grouped node when there are no forks or merges", () => {
65             const commits = [
66                 { id: "1", parentIds: [], repo: "A", branch: "main", date: "2021-01-01", url: "url1" },
67                 { id: "2", parentIds: ["1"], repo: "A", branch: "main", date: "2021-01-02", url: "url2" },
68                 { id: "3", parentIds: ["2"], repo: "A", branch: "main", date: "2021-01-03", url: "url3" },
69             ];
70             const groups = groupNodes(commits);
71             // expect one group covering all commits.
72             expect(groups.length).toBe(1);
73             expect(groups[0].nodes).toEqual(["1", "2", "3"]);
74             expect(groups[0].repo).toBe("A");
75             expect(groups[0].date).toBe("2021-01-01");
76             expect(groups[0].end_date).toBe("2021-01-03");
77             // for default grouping, branch should be "default" and url empty.
78             expect(groups[0].branch).toBe("default");
79             expect(groups[0].url).toBe("");
80         });
81     
82         it("should create separate groups for fork and merge commits", () => {
83             const commits = [
84                 { id: "1", parentIds: [], repo: "A", branch: "main", date: "2021-01-01", url: "url1" },
85                 { id: "2", parentIds: ["1"], repo: "A", branch: "main", date: "2021-01-02", url: "url2" },
86                 { id: "3", parentIds: ["2"], repo: "B", branch: "main", date: "2021-01-03", url: "url3" },
87                 { id: "4", parentIds: ["3", "2"], repo: "B", branch: "main", date: "2021-01-04", url: "url4" },
88             ];
89             const groups = groupNodes(commits);
90             // expect more than one group because of the fork and merge events.
91             expect(groups.length).toBeGreaterThan(1);
92             // find at least one grouped node with branch type "merge" or "forkParent"
93             const specialGroup = groups.find(
94                 (group) => group.branch === "merge" || group.branch === "forkParent"
95             );
96             expect(specialGroup).toBeDefined();
97     
98             // verify that all commit ids appear in one of the groups.
99             const allCommitIds = groups.flatMap((group) => group.nodes);
100             expect(allCommitIds.sort()).toEqual(["1", "2", "3", "4"].sort());
101     
102             // check that special groups have parentIds assigned.
103             if (specialGroup) {
104                 expect(specialGroup.parentIds.length).toBeGreaterThan(0);
105             }
106         });
