34      * Build a Map<repo, color> using d3.interpolateRainbow.
35      * Used to color-code nodes belonging to different forks.
36      */
37     export function buildRepoColorMap(commits: Commit[]): Map<string, string> {
38         const repos = Array.from(new Set(commits.map(c => c.repo)));
39         const map = new Map<string, string>();
40       
41         repos.forEach((repo, i) => {
42             map.set(repo, interpolateRainbow(i / repos.length));
43         });
44       
45         return map;
46     }

60     export function groupBy<T, K>(items: T[], keyFn: (item: T) => K): Map<K, T[]> {
61         const map = new Map<K, T[]>();
62         for (const item of items) {
63             const key = keyFn(item);
64             if (!map.has(key)) {
65                 map.set(key, []);
66             }
67             map.get(key)?.push(item);
68         }
69         return map;
70     }

72     /**
73      * Overwrites y-coordinates to place commits from different
74      * forks in their own unique lanes.
75      * 
76      * @param nodes the data as nodes returned by d3dag builder
77      * 
78      */
79     export function assignUniqueLanes(
80         nodes: Iterable<GraphNode<Commit | GroupedNode, unknown>>
81     ) {
82         // group nodes by repo.
83         const repoGroups = groupBy(Array.from(nodes), (node) => node.data.repo);
84     
85         // sort repos by the date of their earliest commit.
86         const repoOrder = Array.from(repoGroups.entries()).sort((a, b) => {
87             const earliestA =
88                     min(a[1].map((n) => new Date(n.data.date).getTime())) || 0;
89             const earliestB =
90                     min(b[1].map((n) => new Date(n.data.date).getTime())) || 0;
91             return earliestA - earliestB;
92         });
93     
94         let cumulativeOffset = c.MARGIN.top; // leaves space on top
95         const lanes: Record<string, { minY: number; maxY: number; }> = {};
96     
97         // shift nodes for each repo
98         repoOrder.forEach(([repo, repoNodes]) => {
99             const minY = min(repoNodes, (n) => n.y) || 0;
100             const maxY = max(repoNodes, (n) => n.y) || 0;
101             const height = maxY - minY;
102             repoNodes.forEach((node) => {
103                 node.y = cumulativeOffset + (node.y - minY);
104             });
105             lanes[repo] = { minY: cumulativeOffset, maxY: cumulativeOffset + height };
106             cumulativeOffset += height + c.NODE_RADIUS * 2;
107         });
108     
109         return { lanes, totalHeight: cumulativeOffset };
110     }

112     function topologicalSort(commits: Commit[]): Commit[] {
113         const sortedCommits: Commit[] = [];
114         const visited = new Set<string>();
115     
116         const commitMap = new Map(commits.map(commit => [commit.id, commit]));
117     
118         function visit(commit: Commit) {
119             if (!visited.has(commit.id)) {
120                 visited.add(commit.id);
121                 for (const parentId of commit.parentIds) {
122                     const parentCommit = commitMap.get(parentId);
123                     if (parentCommit) { // if a parent commit exists
124                         visit(parentCommit);
125                     }
126                 }
127                 sortedCommits.push(commit);
128             }
129         }
130     
131         // sort commits initially by date
132         const dateSortedCommits = commits.slice().sort(
133             (a, b) => new Date(a.date).getTime() - new Date(b.date).getTime()
134         );
135     
136         // topological sorting
137         for (const commit of dateSortedCommits) {
138             visit(commit);
139         }
140     
141         return sortedCommits;
142     }

144     export function groupNodes(data: Commit[]): GroupedNode[] {
145         const sortedCommits = topologicalSort(data);
146     
147         const groupedNodes: GroupedNode[] = [];
148         const commitLookup = new Map<string, Commit>();
149     
150         const forkParentIds = new Set<string>();
151         const forkParentChildren = new Set<string>(); // commits consisting of pulling from another repo
152         const mergeNodes = new Set<string>();
153         let counter = 0; // used for setting IDs
154     
155         // Find fork parents and merge nodes
156         sortedCommits.forEach(commit => {
157             commitLookup.set(commit.id, commit);
158             // fork parents have a child within a different repo
159             commit.parentIds.forEach(parentId => {
160                 const parentCommit = commitLookup.get(parentId);
161                 if (parentCommit && parentCommit.repo !== commit.repo) {
162                     forkParentIds.add(parentId);
163                     forkParentChildren.add(commit.id);
164                 }
165             });
166             if (
167                 // merge nodes have at least two parents, one from different repo
168                 commit.parentIds.length >= 2 &&
169                 commit.parentIds.some(parentId => {
170                     const parentCommit = commitLookup.get(parentId);
171                     return parentCommit !== undefined && parentCommit.repo !== commit.repo;
172                 })
173             ) {
174                 mergeNodes.add(commit.id);
175             }
176         });
177     
178         const repoGroups = groupBy(Array.from(sortedCommits), (commit) => commit.repo);
179     
180         // Helper function to create a grouped node
181         function createGroupedNode(
182             nodes: Commit[], 
183             startIdx: number, 
184             endIdx: number,
185             branchType = "default", 
186             isSpecial = false): GroupedNode {
187             const nodeIds = nodes.slice(startIdx, endIdx).map(node => node.id);
188             const prefix = isSpecial ? "Special " : "";
189             return {
190                 id: `${prefix}${counter++}`,
191                 parentIds: [],
192                 repo: nodes[startIdx].repo,
193                 branch: branchType,
194                 date: nodes[startIdx].date,
195                 url: branchType === "default" ? "" : nodes[startIdx].url, // Only special nodes are clickable
196                 nodes: nodeIds,
197                 end_date: nodes[endIdx - 1].date,
198             };
199         }
200     
201         // Process each repo group
202         repoGroups.forEach((nodes) => {
203             let lastBreak = 0;
204             
205             for (let i = 0; i < nodes.length; i++) {
206                 const nodeId = nodes[i].id;
207                 const isSpecialNode = mergeNodes.has(nodeId) || forkParentIds.has(nodeId);
208                 const isForkChild = forkParentChildren.has(nodeId);
209                 
210                 if (isSpecialNode || isForkChild) {
211                     // create a group for any nodes between the last break and current special node
212                     if (lastBreak < i) {
213                         groupedNodes.push(createGroupedNode(nodes, lastBreak, i));
214                     }
215                     
216                     // create appropriate node based on type
217                     if (isSpecialNode) {
218                         const type = mergeNodes.has(nodeId) ? "merge" : "forkParent";
219                         groupedNodes.push(createGroupedNode(nodes, i, i + 1, type, true));
220                     } else if (isForkChild) {
221                         // create a default node for pulling from different repo
222                         groupedNodes.push(createGroupedNode(nodes, i, i + 1));
223                     }
224                     
225                     lastBreak = i + 1;
226                 }
227             }
228             
229             // Handle any remaining nodes
230             if (lastBreak < nodes.length) {
231                 groupedNodes.push(createGroupedNode(nodes, lastBreak, nodes.length));
232             }
233         });
234     
235         // Create mapping from commit ID to group ID for parent lookup
236         const commitIdToGroupId = new Map<string, string>();
237         groupedNodes.forEach(group => {
238             group.nodes.forEach(commitId => commitIdToGroupId.set(commitId, group.id));
239         });
240     
241         function findParent(node: GroupedNode) {
242             const firstCommit = commitLookup.get(node.nodes[0]);
243             if (firstCommit) {
244                 firstCommit.parentIds.forEach(parentId => {
245                     const parentGroupId = commitIdToGroupId.get(parentId);
246                     if (parentGroupId) {
247                         node.parentIds.push(parentGroupId);
248                     }
249                 });
250             }
251         }
252     
253         // Assign parents to all groups
254         for (const node of groupedNodes) {
255             findParent(node);
256         }
257     
258         return groupedNodes;
259     }
