1     import { useEffect, useRef, useMemo, useCallback, useState, memo } from "react";
2     import { select } from "d3-selection";
3     import { scaleBand, scaleLinear } from "d3-scale";
4     import { max } from "d3-array";
5     import { timeFormat } from "d3-time-format";
6     import { brushX } from "d3-brush";
7     import { axisBottom } from "d3-axis";
8     import { useTheme, themeGet } from "@primer/react";
9     import { HistogramData } from "@VisInterfaces/HistogramData";
10     import { sortDates, computeFrequency } from "./HistogramUtils";
11     
12     /**
13      * Component that renders a bar chart using D3 with brush selection functionality.
14      */
15     function Histogram({ commitData, handleHistogramSelection }: HistogramData) {
16         const svgRef = useRef<SVGSVGElement | null>(null);
17         const { theme } = useTheme();
18         const [startLabel, setStartLabel] = useState("");
19         const [endLabel, setEndLabel] = useState("");
20     
21         // Memoize bar colors
22         const barColors = useMemo(() => {
23             return {
24                 bgColor: themeGet("colors.neutral.subtle")({ theme }),
25                 bgColorSelected: themeGet("colors.neutral.emphasis")({ theme }),
26                 barColor: themeGet("colors.accent.muted")({ theme }),
27                 barColorSelected: themeGet("colors.accent.emphasis")({ theme }),
28             };
29         }, [theme]);
30     
31         // Extract and sort commit dates
32         const dates = useMemo(() => sortDates(commitData), [commitData]);
33     
34         // Compute frequency of dates
35         const frequency = useMemo(() => computeFrequency(dates), [dates]);
36     
37         // Memoize chart dimensions
38         const chartDimensions = useMemo(() => {
39             const height = 300;
40             const focusContextRatio = 0.7;
41     
42             const contextMargin = { top: 0, right: 10, bottom: 45, left: 10 };
43             const contextHeight =
44                 (1 - focusContextRatio) * height - contextMargin.top - contextMargin.bottom;
45     
46             const focusMargin = { top: 0, right: 10, bottom: 10, left: 10 };
47             const focusHeight =
48                 focusContextRatio * height - focusMargin.top - focusMargin.bottom;
49     
50             return { height, focusContextRatio, contextMargin, contextHeight, focusMargin, focusHeight };
51         }, []);
52     
53         // Draw the chart
54         const drawChart = useCallback(() => {
55             if (!svgRef.current) return;
56     
57             // Setup chart dimensions
58             const svg = select(svgRef.current);
59             const container = svg.node()?.parentElement as HTMLElement;
60             const width = container.clientWidth;
61     
62             const {
63                 height,
64                 contextMargin,
65                 contextHeight,
66                 focusMargin,
67                 focusHeight,
68             } = chartDimensions;
69     
70             const contextWidth = width - contextMargin.left - contextMargin.right;
71             const focusWidth = width - focusMargin.left - focusMargin.right;
72     
73             // Define scales for context chart
74             const formattedDates = Array.from(frequency.keys()).map((dateStr) =>
75                 timeFormat("%b %Y")(new Date(dateStr))
76             );
77     
78             const xScaleContext = scaleBand()
79                 .domain(formattedDates)
80                 .range([0, contextWidth])
81                 .padding(0.1);
82             const yScaleContext = scaleLinear()
83                 .domain([0, max(Array.from(frequency.values())) || 1])
84                 .range([contextHeight, 0]);
85     
86             // Define scales for focus chart
87             let xScaleFocus = scaleBand()
88                 .domain(formattedDates)
89                 .range([0, focusWidth])
90                 .padding(0.1);
91             const yScaleFocus = scaleLinear()
92                 .domain([0, max(Array.from(frequency.values())) || 1])
93                 .range([focusHeight, 0]);
94     
95             // Set SVG size
96             svg
97                 .attr("width", width)
98                 .attr("height", height)
99                 .attr("viewBox", `0 0 ${width} ${height}`);
100             svg.selectAll("*").remove();
101     
102             // Initialize context chart
103             const chartContext = svg
104                 .append("g")
105                 .attr("class", "chartContext")
106                 .attr(
107                     "transform",
108                     `translate(${contextMargin.left},
109                 ${focusHeight +
110                     focusMargin.top +
111                     focusMargin.bottom +
112                     contextMargin.top})`
113                 );
114     
115             // Draw background bars for context chart
116             chartContext
117                 .selectAll(".bar-bg")
118                 .data(Array.from(frequency))
119                 .enter()
120                 .append("rect")
121                 .attr("class", "bar-bg")
122                 .attr("x", (d) => xScaleContext(timeFormat("%b %Y")(new Date(d[0]))) || 0)
123                 .attr("y", 0)
124                 .attr("width", xScaleContext.bandwidth())
125                 .attr("height", contextHeight)
126                 .style("fill", barColors.bgColor);
127     
128             // Draw bars for context chart
129             chartContext
130                 .selectAll(".bar")
131                 .data(Array.from(frequency))
132                 .enter()
133                 .append("rect")
134                 .attr("class", "bar")
135                 .attr("x", (d) => xScaleContext(timeFormat("%b %Y")(new Date(d[0]))) || 0)
136                 .attr("y", (d) => yScaleContext(d[1]))
137                 .attr("width", xScaleContext.bandwidth())
138                 .attr("height", (d) => contextHeight - yScaleContext(d[1]))
139                 .style("fill", barColors.barColor);
140     
141             // Brush setup
142             const brush = brushX()
143                 .extent([
144                     [0, 0],
145                     [contextWidth, contextHeight],
146                 ])
147                 .on("end", ({ selection }) => {
148                     if (selection) {
149                         const [x0, x1] = selection;
150                         // Update background bar colors based on selection
151                         chartContext.selectAll(".bar-bg").style("fill", (d: unknown) => {
152                             const data = d as [string, number];
153                             const pos = xScaleContext(
154                                 timeFormat("%b %Y")(new Date(data[0]))
155                             ) as number;
156                             return pos >= x0 && pos <= x1
157                                 ? barColors.bgColorSelected
158                                 : barColors.bgColor;
159                         });
160     
161                         // Update bar colors based on selection
162                         chartContext.selectAll(".bar").style("fill", (d: unknown) => {
163                             const data = d as [string, number];
164                             const pos = xScaleContext(
165                                 timeFormat("%b %Y")(new Date(data[0]))
166                             ) as number;
167                             return pos >= x0 && pos <= x1
168                                 ? barColors.barColorSelected
169                                 : barColors.barColor;
170                         });
171     
172                         // Store selected date range
173                         const selectedDates = Array.from(frequency).filter((d) => {
174                             const pos = xScaleContext(
175                                 timeFormat("%b %Y")(new Date(d[0]))
176                             ) as number;
177                             return pos >= x0 && pos <= x1;
178                         });
179     
180                         // Update focus chart
181                         xScaleFocus = scaleBand()
182                             .domain(
183                                 selectedDates.map((d) => timeFormat("%b %Y")(new Date(d[0])))
184                             )
185                             .range([0, focusWidth])
186                             .padding(0.1);
187                         chartFocus.selectAll("*").remove();
188     
189                         // Draw background bars for focus chart
190                         chartFocus
191                             .selectAll(".bar-bg")
192                             .data(Array.from(selectedDates))
193                             .enter()
194                             .append("rect")
195                             .attr("class", "bar-bg")
196                             .attr(
197                                 "x",
198                                 (d) => xScaleFocus(timeFormat("%b %Y")(new Date(d[0]))) || 0
199                             )
200                             .attr("y", 0)
201                             .attr("width", xScaleFocus.bandwidth())
202                             .attr("height", focusHeight)
203                             .style("fill", barColors.bgColor);
204     
205                         // Draw bars for focus chart
206                         chartFocus
207                             .selectAll(".bar")
208                             .data(Array.from(selectedDates))
209                             .enter()
210                             .append("rect")
211                             .attr("class", "bar")
212                             .attr(
213                                 "x",
214                                 (d) => xScaleFocus(timeFormat("%b %Y")(new Date(d[0]))) || 0
215                             )
216                             .attr("y", (d) => yScaleFocus(d[1]))
217                             .attr("width", xScaleFocus.bandwidth())
218                             .attr("height", (d) => focusHeight - yScaleFocus(d[1]))
219                             .style("fill", barColors.barColorSelected)
220                             // Tooltip event handlers
221                             .on("mouseover", function () {
222                                 tooltip.style("opacity", 1);
223                             })
224                             .on("mousemove", function (event, d) {
225                                 tooltip
226                                     .html(
227                                         `Date: ${timeFormat("%b %Y")(new Date(d[0]))}
228                                         <br/>Commits: ${d[1]}`
229                                     )
230                                     .style("left", event.pageX + 10 + "px")
231                                     .style("top", event.pageY + 10 + "px");
232                             })
233                             .on("mouseout", function () {
234                                 tooltip.style("opacity", 0);
235                             });
236     
237                         chartFocus.selectAll(".selection-label").remove();
238     
239                         if (selectedDates.length > 0) {
240                             // Get start and end date
241                             const dateFormat = timeFormat("%B %Y");
242                             const startOfSelection = new Date(selectedDates[0]?.[0]);
243     
244                             const endDateStr = selectedDates[selectedDates.length - 1][0];
245                             const endDateObj = new Date(endDateStr);
246                             const endOfSelection = new Date(
247                                 endDateObj.getFullYear(),
248                                 endDateObj.getMonth() + 1,
249                                 0, // Last day of the month
250                                 23,
251                                 59,
252                                 59,
253                                 999 // Set to the latest possible time
254                             );
255     
256                             if (startOfSelection && endOfSelection) {
257                                 setStartLabel(dateFormat(startOfSelection));
258                                 setEndLabel(dateFormat(endOfSelection));
259                                 if (handleHistogramSelection)
260                                     handleHistogramSelection(startOfSelection, endOfSelection);
261                             }
262                         }
263                     }
264                 });
265     
266             // Define space for x-axis labels
267             const labelWidth = 30;
268             const tickCount = Math.floor(contextWidth / labelWidth);
269             const skip = Math.max(1, Math.ceil(formattedDates.length / tickCount));
270     
271             // Draw x-axis labels
272             chartContext
273                 .append("g")
274                 .attr("transform", `translate(0, ${contextHeight})`)
275                 .call(
276                     axisBottom(xScaleContext)
277                         .tickValues(formattedDates.filter((_, i) => i % skip === 0))
278                         .tickFormat((d) => d)
279                 )
280                 .selectAll("text")
281                 .style("text-anchor", "end")
282                 .attr("dx", "-0.8em")
283                 .attr("dy", "0.15em")
284                 .attr("transform", "rotate(-45)");
285     
286             // Draw bars for focus chart
287             const chartFocus = svg
288                 .append("g")
289                 .attr("class", "chartFocus")
290                 .attr("transform", `translate(${focusMargin.left},${focusMargin.top})`);
291     
292             // Append brush to context chart
293             chartContext
294                 .append("g")
295                 .attr("class", "brush")
296                 .call(brush)
297                 .call(brush.move, [0, contextWidth / 5]);
298     
299             // Tooltip setup
300             const tooltip = select("body")
301                 .append("div")
302                 .attr("class", "tooltip")
303                 .style("position", "absolute")
304                 .style("padding", "6px")
305                 .style("background", "#f9f9f9")
306                 .style("border", "1px solid #ccc")
307                 .style("border-radius", "4px")
308                 .style("pointer-events", "none")
309                 .style("opacity", 0);
310         }, [frequency]);
311     
312         useEffect(() => {
313             drawChart();
314             window.addEventListener("resize", drawChart);
315             return () => window.removeEventListener("resize", drawChart);
316         }, [drawChart, handleHistogramSelection, chartDimensions]);
317     
318         return (
319             <div style={{ borderRadius: "10px" }}>
320                 <div style={{ textAlign: "center", marginBottom: "0.5rem" }}>
321                     <h1 style={{ fontSize: "16px", fontWeight: "bold", margin: 0 }}>
322                         Slide to select a date range
323                     </h1>
324                 </div>
325                 <div
326                     style={{
327                         display: "flex",
328                         justifyContent: "space-between",
329                         alignItems: "center",
330                         marginBottom: "0.5rem",
331                     }}
332                 >
333                     <span style={{ fontWeight: "bold", fontSize: "14px" }}>
334                         {startLabel}
335                     </span>
336                     <span style={{ fontWeight: "bold", fontSize: "14px" }}>{endLabel}</span>
337                 </div>
338                 <svg
339                     ref={svgRef}
340                     style={{
341                         width: "100%",
342                         height: "100%",
343                         borderRadius: "8px",
344                     }}
345                 ></svg>
346             </div>
347         );
348     };
349     
350     export default memo(Histogram);
