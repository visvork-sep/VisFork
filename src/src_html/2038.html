1     import { utcMonth, utcMonths } from "d3-time";
2     /*
3       * Sorts an array of commit data objects by their date property.
4       * Returns A sorted array of dates.
5       */
6     export function sortDates(commitData: { date: Date }[]) {
7         const sortedCommits = [...commitData].sort(
8             (a, b) => a.date.getTime() - b.date.getTime()
9         );
10         return sortedCommits.map((commit) => commit.date);
11     }

13     /*
14       * Computes the frequency of commits per month from a list of dates.
15       * Returns A Map where keys are month strings (ISO format) and values are commit counts.
16       */
17     export function computeFrequency(dates: Date[]) {
18         // Convert dates to UTC months and then to ISO strings for map keys
19         const monthStrings = dates.map((d) => utcMonth(d).toISOString());
20       
21         // Create frequency map with string keys
22         const freqMap = new Map<string, number>();
23         monthStrings.forEach((key) => {
24             freqMap.set(key, (freqMap.get(key) ?? 0) + 1);
25         });
26     
27         // If we have dates, fill in missing months
28         if (dates.length > 0) {
29         // Get min and max dates
30             const minDate = utcMonth(new Date(Math.min(...monthStrings.map(str => new Date(str).getTime()))));
31             const maxDate = utcMonth(new Date(Math.max(...monthStrings.map(str => new Date(str).getTime()))));
32         
33             // Fill gaps with zeros
34             utcMonths(minDate, maxDate).forEach((date) => {
35                 const key = date.toISOString();
36                 if (!freqMap.has(key)) {
37                     freqMap.set(key, 0);
38                 }
39             });
40         }
41     
42         // Sort by date (parsing strings back to dates for comparison)
43         return new Map(
44             Array.from(freqMap).sort((a, b) => {
45                 const dateA = new Date(a[0]);
46                 const dateB = new Date(b[0]);
47                 return dateA.getTime() - dateB.getTime();
48             })
49         );
50     }
