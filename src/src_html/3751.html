20     function useFormSubmission(form: FilterFormState, onFiltersChange: FilterChangeHandler) {
21         const [repositoryInputError, setRepositoryInputError] = useState<InputError | null>(null);
22         const [forksCountInputError, setForksCountInputError] = useState<InputError | null>(null);
23         const [recentlyUpdatedInputError, setRecentlyUpdatedInputError] =
24             useState<InputError | null>(null);
25         const [commitsDateRangeFromInputError, setCommitsDateRangeFromInputError] =
26             useState<InputError | null>(null);
27         const [commitsDateRangeUntilInputError, setCommitsDateRangeUntilInputError]
28             = useState<InputError | null>(null);
29         const [commitsTypeFilterInputError, setCommitsTypeFilterInputError] = useState<InputError | null>(null);
30         const [ownerTypeFilterInputError, setOwnerTypeFilterInputError] = useState<InputError | null>(null);
31         const [forksOrderInputError, setForksOrderInputError] = useState<InputError | null>(null);
32         const [forksAscDescInputError, setForksAscDescInputError] = useState<InputError | null>(null);
33     
34         /**
35          * Prepares the form state for submission.
36          * Any errors in the format of the form will be caught and the error will be set.
37          * 
38          * If any error is caught, that field will be set to null.
39          * 
40          * @returns {preparedForm} The prepared form state.
41          */
42         const prepareForm = (): preparedForm => {
43     
44             const output = safePrepare(prepareRepository, form.repository, setRepositoryInputError);
45             const { owner, repositoryName } = output ?? { owner: null, repositoryName: null };
46     
47             const forksCount = safePrepare(prepareForksCount, form.forksCount, setForksCountInputError);
48     
49             const forksOrder = safePrepare(prepareForksOrder, form.forksOrder, setForksOrderInputError);
50     
51             const forksSortDirection = safePrepare(prepareForksSortDirection, form.forksAscDesc, setForksAscDescInputError);
52     
53             let commitsDateRangeFrom =
54                 safePrepare(prepareCommitsDateRangeFrom, form.commitsDateRangeFrom, setCommitsDateRangeFromInputError);
55     
56             let commitsDateRangeUntil =
57                 safePrepare(prepareCommitsDateRangeUntil, form.commitsDateRangeUntil, setCommitsDateRangeUntilInputError);
58     
59             const commitsTypeFilter =
60                 safePrepare(prepareCommitsTypeFilter, form.commitTypeFilter, setCommitsTypeFilterInputError);
61     
62             const ownerTypeFilter = safePrepare(prepareOwnerTypeFilter, form.ownerTypeFilter, setOwnerTypeFilterInputError);
63     
64             const recentlyUpdated = safePrepare(prepareRecentlyUpdated, form.recentlyUpdated, setRecentlyUpdatedInputError);
65             if (commitsDateRangeFrom && commitsDateRangeUntil && commitsDateRangeFrom > commitsDateRangeUntil) {
66                 setCommitsDateRangeFromInputError(new CommitsDateRangeFromInputErrors.LaterFromDateError());
67                 setCommitsDateRangeUntilInputError(new CommitsDateRangeFromInputErrors.LaterFromDateError());
68                 commitsDateRangeFrom = null;
69                 commitsDateRangeUntil = null;
70             }
71     
72             return {
73                 owner,
74                 repositoryName,
75                 forksCount,
76                 forksOrder,
77                 forksSortDirection,
78                 commitsDateRangeFrom,
79                 commitsDateRangeUntil,
80                 commitsTypeFilter,
81                 ownerTypeFilter,
82                 recentlyUpdated
83             };
84         };
85     
86         const onSubmit = (event: FormEvent) => {
87             // Prevents the page from refreshing on submission
88             event.preventDefault();
89     
90             const preparedForm = prepareForm();
91     
92             // If any of the required fields are null, the form is not ready for submission
93             if (
94                 !preparedForm.owner
95                 || !preparedForm.repositoryName
96                 || !preparedForm.forksCount
97                 || !preparedForm.forksOrder
98                 || !preparedForm.forksSortDirection
99                 || !preparedForm.commitsDateRangeFrom
100                 || !preparedForm.commitsDateRangeUntil
101                 || !preparedForm.commitsTypeFilter
102                 || !preparedForm.ownerTypeFilter
103             ) {
104                 return;
105             }
106     
107             const completeForm = preparedForm as preparedFormComplete;
108     
109             // Type assertion because we know that the form is complete
110             onFiltersChange(filterFactory(completeForm), forkQueryStateFactory(completeForm));
111         };
112     
113         return {
114             onSubmit,
115             repositoryInputError,
116             forksCountInputError,
117             commitsDateRangeFromInputError,
118             commitsDateRangeUntilInputError,
119             recentlyUpdatedInputError,
120             commitsTypeFilterInputError,
121             ownerTypeFilterInputError,
122             forksOrderInputError,
123             forksAscDescInputError
124         };
125     }
126     
127     export {
