1     import { SplitPageLayout, Dialog, Button } from "@primer/react";
2     import { useLocation, useNavigate } from "react-router-dom";
3     import { useEffect, useState } from "react";
4     import AppHeader from "@Components/AppHeader";
5     import DataComponents from "@Components/DataComponents";
6     
7     /**
8      * HomePage Component
9      *
10      * Displays the main layout of the application, including:
11      * - A header with navigation and authentication controls.
12      * - The primary data display component (DataComponents).
13      * - A footer for the page.
14      * - An optional error popup if the user encounters an OAuth error.
15      */
16     export default function HomePage() {
17         // Gives information about the current route, including any state passed via navigation.
18         const location = useLocation();
19         // Allows programmatic navigation to other routes in the application.
20         const navigate = useNavigate();
21     
22         //Determines whether an error dialog should appear, based on the router state.
23         //We store it in local state so that once we clear the router state, this flag won't reset.
24         const [isErrorPopupOpen, setIsErrorPopupOpen] = useState(() => {
25             return location.state?.fromError ?? false;
26         });
27         
28         //Holds the error text we want to display in the dialog, if any.
29         //Copied from location.state, ensuring we keep the message after clearing router state.
30         const [errorText] = useState(() => {
31             return location.state?.errorDescription ?? "";
32         });
33     
34         //Clears location.state after the first render if an error was present.
35         //This prevents the error dialog from reopening on a page refresh.
36         useEffect(() => {
37             if (location.state?.fromError) {
38                 navigate(".", { replace: true, state: {} });
39             }
40         }, [location, navigate]);
41     
42         return (
43             <>
44                 {/* 
45                     Primary layout container using Primerâ€™s SplitPageLayout component.
46                     Includes a header, main section, and a footer. 
47                 */}
48                 <SplitPageLayout>
49                     <SplitPageLayout.Header aria-label="Header">
50                         <AppHeader />
51                     </SplitPageLayout.Header>
52                     <DataComponents />
53                     <SplitPageLayout.Footer aria-label="Footer">
54                         <div>Footer</div>
55                     </SplitPageLayout.Footer>
56                 </SplitPageLayout>
57                 
58                 {/**
59                     Conditionally render the Dialog if an error was detected.
60                 */}
61                 {isErrorPopupOpen && (
62                     <Dialog
63                         title="Authentication Error"
64                         onClose={() => setIsErrorPopupOpen(false)}
65                         position="center"
66                     >   
67                         <Dialog.Body>
68                             <p>Authentication ran into an error:  <br />{errorText || "No error message provided."} </p>
69                             <Button onClick={() => setIsErrorPopupOpen(false)}>
70                                 Close
71                             </Button>
72                         </Dialog.Body>
73                     </Dialog>
74                 )}
75             </>
76         );
77     }
