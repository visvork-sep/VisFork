1     import { Box, Text, Select } from "@primer/react";
2     import { DataTable, Table } from "@primer/react/experimental";
3     import { useState, useMemo, memo } from "react";
4     import { ForkListData, ForkListDetails } from "@VisInterfaces/ForkListData";
5     
6     function ForkList({ forkData }: ForkListData) {
7         // States to control page size and current page index
8         const [pageSize, setPageSize] = useState(10);
9         const [pageIndex, setPageIndex] = useState(0);
10     
11         // Memoize the items to display on the current page
12         const forksToShow = useMemo(() => {
13             const start = pageIndex * pageSize;
14             const end = start + pageSize;
15             return forkData.slice(start, end);
16         }, [forkData, pageSize, pageIndex]);
17     
18         // Memoize column definitions
19         const dataColumns = useMemo(() => [
20             { header: "Name", field: "name" as keyof ForkListDetails, rowHeader: true },
21             { header: "Description", field: "description" as keyof ForkListDetails, rowHeader: true },
22         ], []);
23     
24         return (
25             <Table.Container>
26                 <Table.Title as="h2" id="forks">Forks</Table.Title>
27     
28                 <Table.Actions>
29                     <Box display="flex" alignItems="center">
30                         <Text as="span" size="medium" weight="semibold" padding="1">
31                             Rows per page
32                         </Text>
33                         <Select
34                             size="medium"
35                             onChange={(e) => {
36                                 setPageIndex(0);
37                                 setPageSize(Number(e.target.value));
38                             }}
39                             defaultValue={pageSize.toString()}
40                         >
41                             {Array.from({ length: 100 }, (_, i) => i + 1).map((value) => (
42                                 <Select.Option key={value} value={value.toString()}>
43                                     {value}
44                                 </Select.Option>
45                             ))}
46                         </Select>
47                     </Box>
48                 </Table.Actions>
49     
50                 <Table.Subtitle as="p" id="forks-subtitle">
51                     A list of all the forks of the submitted repository.
52                 </Table.Subtitle>
53     
54                 <DataTable
55                     aria-labelledby="forks"
56                     aria-describedby="forks-subtitle"
57                     data={forksToShow}
58                     columns={dataColumns}
59                 />
60     
61                 <Table.Pagination
62                     key={pageSize}
63                     aria-label="Pagination for Forks"
64                     pageSize={pageSize}
65                     totalCount={forkData.length}
66                     onChange={({ pageIndex }) => setPageIndex(pageIndex)}
67                 />
68             </Table.Container>
69         );
70     }
71     
72     export default memo(ForkList);
