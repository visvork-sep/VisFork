25     /**
26      * Fetches fork data using a the underlying Json function as well as the original repo data.
27      * Converts the ForkQueryState to the necessary query parameters.
28      *
29      * @param forkQueryState - ForkQuery state, high level representation of the settings for querrying.
30      * If required params are empty query will be disabled.
31      * @returns The result object is of type ForkInfo.
32      */
33     export function useFetchForks(forkQueryState?: ForkQueryState) {
34         const { isAuthenticated, getAccessToken } = useAuth();
35         const forkQueryParams: ForkQueryParams | undefined = forkQueryState
36             ? createForkQueryParams(forkQueryState.owner, forkQueryState.repo, forkQueryState.sort)
37             : undefined;
38     
39         const accessToken = getAccessToken() ?? "";
40     
41         return useQuery({
42             queryKey: ["forks", forkQueryState],
43             queryFn: () => fetchForks(forkQueryParams as ForkQueryParams, accessToken, forkQueryState?.forksCount),
44             // Done so query is not triggered on first render.
45             enabled: isAuthenticated && !!forkQueryParams,
46             refetchOnWindowFocus: false,
47         });
48     }

50     /**
51      * Fetches commit data using a the underlying Json function.
52      * Takes the forks to query commits for and returns an array of commits.
53      *
54      * @param parameters - An array of forks and a data range for which to query.
55      * @returns The result object is of type CommitInfo.
56      */
57     
58     export function useFetchCommitsBatch(forks: UnprocessedRepository[], forkQueryState?: ForkQueryState) {
59         const { isAuthenticated, getAccessToken } = useAuth();
60         const accessToken = getAccessToken() ?? "";
61     
62     
63         const commitQueryParameters: CommitQueryParams[] = useMemo(() => forkQueryState ? forks.map(fork =>
64             createCommitQueryParams(
65                 fork.owner.login,
66                 fork.name,
67                 forkQueryState.range.start.toISOString().split("T")[0],
68                 forkQueryState.range.end.toISOString().split("T")[0]
69             ))
70             : [], [forks]);
71     
72         // Check if main repository has a lot of commits
73         useEffect(() => {
74             if (commitQueryParameters.length > 0) {
75                 fetchCommitCount(commitQueryParameters[0], accessToken).then((commitsCount) => {
76                     if (commitsCount > 2000) {
77                         alert("Repository contains a lot of commits for this time range. This might take a while.");
78                     }
79                 });
80             }
81         }, [commitQueryParameters]);
82     
83         return useQueries({
84             queries: commitQueryParameters.map((parameters) => {
85                 return {
86                     queryKey: ["commits", parameters],
87                     queryFn: async () => fetchCommits(parameters, accessToken),
88                     enabled: isAuthenticated,
89                     refetchOnWindowFocus: false,
90                 };
91             }),
92         });
93     }
