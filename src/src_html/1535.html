27     export const advanceAutoPlay = (isPlaying: boolean,
28         allDates: string[],
29         currentDateIndex: number,
30         setCurrentDateIndex: React.Dispatch<React.SetStateAction<number>>,
31         playInterval: React.RefObject<NodeJS.Timeout | null>,
32         setIsPlaying: React.Dispatch<React.SetStateAction<boolean>>,
33         playSpeed: number) => {
34         // Currently advances on a 100ms interval
35         if (isPlaying) {
36             if (currentDateIndex === allDates.length - 1) {
37                 setCurrentDateIndex(0);
38             }
39             playInterval.current = setInterval(() => {
40                 setCurrentDateIndex((prev) => {
41                     if (prev < allDates.length - 1) {
42                         return prev + 1;
43                     } else {
44                         setIsPlaying(!isPlaying);
45                         return allDates.length - 1;
46                     }
47                 }
48                 );
49             }, 500 / playSpeed);
50         } else {
51             if (playInterval.current) {
52                 clearInterval(playInterval.current);
53             }
54         }
55     
56         return () => {
57             if (playInterval.current) {
58                 clearInterval(playInterval.current);
59             }
60         };
61     };

63     export function filterUniques(visibleCommits: CollabGraphDetails[], authors: Set<string>, repos: Set<string>, links: Link[], authorNames: Record<string, string>, authorCommitCounts: Record<string, number>, repoCommitCounts: Record<string, number>) {
64         visibleCommits.forEach((entry) => {
65             authors.add(entry.login); // logins are unique, but author names are not
66             repos.add(entry.repo);
67             links.push({
68                 source: entry.login,
69                 target: entry.repo,
70             });
71     
72             authorNames[entry.login] = entry.author; // Store author names for display
73     
74     
75             // Keep track of commit counts for scaling nodes
76             authorCommitCounts[entry.login] = (authorCommitCounts[entry.login] || 0) + 1;
77             repoCommitCounts[entry.repo] = (repoCommitCounts[entry.repo] || 0) + 1;
78         });
79     }
