1     import { InfoIcon } from "@primer/octicons-react";
2     import { Dialog, Button } from "@primer/react";
3     import { useState } from "react";
4     import Tooltip from "@Components/Tooltip/Tooltip";
5     interface InfoButtonProps {
6         title: string; //Name of the visualization
7         shortDescription: string; //Quick hover over description
8         fullDescription: string; // Extended description for clicking
9     }
10     export function InfoButton({ title, shortDescription, fullDescription }: InfoButtonProps) {
11         const [isOpen, setIsOpen] = useState(false);
12     
13         return (
14             <>
15                 <Tooltip text={shortDescription}>
16                     <Button
17                         variant="invisible"
18                         onClick={() => setIsOpen(true)}
19                         aria-label={`More info about ${title}`}
20                     >
21                         <InfoIcon size={16} />
22                     </Button>
23                 </Tooltip>
24     
25                 {isOpen && (
26                     <Dialog title={title} onClose={() => setIsOpen(false)}>
27                         <Dialog.Body>
28                             {fullDescription.split("\n").map((line, index) => (
29                                 <p key={index}>{line}</p>
30                             ))}
31                         </Dialog.Body>
32                     </Dialog>
33                 )}
34             </>
35         );
36     }
