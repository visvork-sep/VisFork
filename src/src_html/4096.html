1     import { createContext, PropsWithChildren, useCallback, useContext, useEffect, useState } from "react";
2     
3     interface AuthContextType {
4         isAuthenticated: boolean;
5         login: (token: string) => void;
6         logout: () => void;
7         getAccessToken: () => string | null;
8     }
9     
10     const AuthContext = createContext<AuthContextType | null>(null);
11     
12     function AuthProvider({ children }: PropsWithChildren) {
13         const [isAuthenticated, setIsAuthenticated] = useState(false);
14     
15         useEffect(() => {
16             // Check for access token in session storage
17             const token = getAccessToken();
18             setIsAuthenticated(!!token);
19         }, []);
20     
21         const login = useCallback((token: string) => {
22             sessionStorage.setItem("access_token", token);
23             setIsAuthenticated(true);
24         }, [isAuthenticated]);
25     
26         const logout = useCallback(() => {
27             sessionStorage.removeItem("access_token");
28             setIsAuthenticated(false);
29         }, [isAuthenticated]);
30     
31         const getAccessToken = useCallback(() => {
32             return sessionStorage.getItem("access_token");
33         }, [isAuthenticated]);
34     
35         return (
36             <AuthContext.Provider value={{ isAuthenticated, login, logout, getAccessToken }}>
37                 {children}
38             </AuthContext.Provider>
39         );
40     };
41     
42     function useAuth() {
43         const context = useContext(AuthContext);
44         if (!context) {
45             throw new Error("Context must be used inside provider");
46         }
47         return context;
48     };
49     
50     export {
51         AuthProvider,
52         useAuth,
53     };
54     export type { AuthContextType };
