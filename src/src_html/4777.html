12     /**
13      * Initializes data structures to be used for processing the commits, after which it calls
14      * {@link removeParentIds} and {@link deleteDuplicateCommits}.
15      * 
16      * @param rawCommits commits array to be processed
17      * @param defaultBranches a map of key-value pairs where the keys are every repo name and the values are
18      * the default branches of those repos. Example (with only 1 element): { "torvalds/linux": "main" }
19      * @param mainRepo the name of the queried repository. Example: torvalds/linux
20      * @returns an array of processed commits where each commit's parentIds array will only contain hashes
21      * of commits present in the array and where no duplicate commits exist
22      */
23     export function processCommits(rawCommits: UnprocessedCommitExtended[],
24         defaultBranches: Record<string, string>,
25         mainRepo: string): UnprocessedCommitExtended[] {
26         // Initialize data structures
27         commitLocationMap.clear();
28         ownerRepoMap.clear();
29         commitMap.clear();
30         locationHeadCommitMap.clear();
31         locationHeadCommitMapReversed.clear();
32         initializeBranchData(defaultBranches, mainRepo);
33         for (const rawCommit of rawCommits) {
34             const locationArray: CommitLocation[] = commitLocationMap.get(rawCommit.sha) ?? [];
35             locationArray.push({ branch: rawCommit.branch as string, repo: rawCommit.repo });
36             commitLocationMap.set(rawCommit.sha, locationArray);
37             commitMap.set(rawCommit.sha, rawCommit);
38             const key = JSON.stringify({ repo: rawCommit.repo, branch: rawCommit.branch });
39             const headCommit = locationHeadCommitMap.get(key);
40             if (rawCommit.date !== "Unknown"
41                 && (headCommit === undefined || (headCommit.date as Date).getTime() < (rawCommit.date as Date).getTime())) {
42                 locationHeadCommitMap.set(
43                     key, rawCommit
44                 );
45             }
46             const owner = rawCommit.repo.split("/")[0];
47             ownerRepoMap.set(owner, rawCommit.repo);
48         }
49         // Reverse locationHeadCommitMap to make it useful for our purpose
50         for (const entry of locationHeadCommitMap.entries()) {
51             const locations: CommitLocation[] = locationHeadCommitMapReversed.get(entry[1].sha) ?? [];
52             locations.push(JSON.parse(entry[0]));
53             locationHeadCommitMapReversed.set(entry[1].sha, locations);
54         }
55     
56         // Remove parentIds that dont exist in data
57         removeParentIds(rawCommits, commitMap);
58     
59         return deleteDuplicateCommits(rawCommits);
60     }

62     /**
63      * Checks for each commit in its input whether it has parentIds that are not in the data.
64      * If it does, it deletes just those parentIds.
65      * 
66      * @param rawCommits commits for the parentIds to be filtered on
67      */
68     function removeParentIds(rawCommits: UnprocessedCommitExtended[], commitMap: Map<string, UnprocessedCommitExtended>) {
69         for (const rawCommit of rawCommits) {
70             for (const parent of rawCommit.parentIds) {
71                 if (!commitMap.get(parent)) {
72                     rawCommit.parentIds = rawCommit.parentIds.filter((p) => p != parent);
73                 }
74             }
75         }
76     }
