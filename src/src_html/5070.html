5     /**
6      * Helper function to create a CommitQueryParams object with defaults.
7      *
8      * @param owner The repository owner
9      * @param repo The repository name
10      * @param since (Optional) Date to fetch commits from
11      * @param until (Optional) Date to fetch commits until
12      * @param perPage (Optional) Number of commits per request (default: 30)
13      * @returns CommitQueryParams object
14      */
15     export function createCommitQueryParams(
16         owner: string,
17         repo: string,
18         since: string,
19         until: string,
20         perPage = 30
21     ): CommitQueryParams {
22         return {
23             path: { owner, repo },
24             query: {
25                 since,
26                 until,
27                 per_page: perPage
28             }
29         };
30     }
