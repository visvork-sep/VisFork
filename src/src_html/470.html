73         const forksOrderInput = useMemo(

75                 <ForksQueryOrderInput

96             () => (

102             ),

104         );
105     
106         const commitTypeFilterInput = useMemo(
107             () => (
108                 <CommitTypeFilterInput
109                     onChangeHandler={handleCommitsTypeFilterChange}

128         const recentlyUpdatedInput = useMemo(

139         return (

144                     <Pagehead>Choose main filters</Pagehead>

146                     <Stack.Item>

173                         </Button>
174                     </Stack.Item>
175                 </Stack>
176             </Box>

178     };
