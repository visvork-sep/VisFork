1     import { ForksSortingOrder, CommitType, OwnerType, SortDirection } from "@Utils/Constants";

3     /**
4      * Represents the state of the filter form used to configure 
5      * the visualization of GitHub repository data.
6      */
7     interface FilterFormState {
8         /**
9          * The full repository name in the format "Owner/RepositoryName".
10          */
11         repository: string;
12     
13         /**
14          * The minimum number of forks required for filtering.
15          * Stored as a string to accommodate user input before validation.
16          */
17         forksCount: string;
18     
19         /**
20          * The selected sorting criteria for forks (e.g., by stargazers, creation date, etc.).
21         * Must be one of the defined sorting criteria in FORKS_SORTING_ORDERS.
22          */
23         forksOrder: ForksSortingOrder;
24     
25         /**
26          * The sorting direction for forks (ascending or descending).
27          * Must be one of the defined sorting directions in SORT_DIRECTION.
28          */
29         forksAscDesc: SortDirection;
30     
31         /**
32          * The starting date for filtering commits.
33          * Expected format: YYYY-MM-DD.
34          */
35         commitsDateRangeFrom: string;
36     
37         /**
38          * The ending date for filtering commits.
39          * Expected format: YYYY-MM-DD.
40          */
41         commitsDateRangeUntil: string;
42     
43         /**
44          * The selected commit types to be included in the visualization.
45          * This is an array of strings representing available commit categories.
46          * Must be one of the defined commit types in COMMIT_TYPES.
47          */
48         commitTypeFilter: CommitType[];
49     
50         /**
51          * The selected owner types to filter repositories by.
52          * This is an array of strings representing different repository ownership categories.
53          * Must be one of the defined owner types in OWNER_TYPES.
54          */
55         ownerTypeFilter: OwnerType[];
56     
57         /**
58          * Specifies the recency of updates to filter repositories by.
59          * Typically represents a number of months, stored as a string.
60          */
61         recentlyUpdated: string;
62     }
63

89         forksOrder: ForksSortingOrder | null;
90         forksSortDirection: SortDirection | null;
91         commitsDateRangeFrom: Date | null;

94         ownerTypeFilter: OwnerType[] | null;
