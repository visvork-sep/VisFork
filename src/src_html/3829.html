4     // Base error class for input validation errors
5     class InputError extends Error {
6         constructor(message: string) {
7             super(message); 
8             this.name = this.constructor.name; 
9         }
10     }
11     
12     // Generic error class for unknown errors
13     class UnknownError extends InputError {
14         constructor() {
15             super("Unknown error"); // Setting a default error message
16         }
17     }
18     
19     class ForbiddenCharactersError extends InputError {
20         constructor(forbidden: string[]) {
21             super("Forbidden characters: " + forbidden.toLocaleString);
22         }
23     }
24     
25     // Error for invalid date range where 'from' date is later than 'until' date
26     class LaterFromDateError extends InputError {
27         constructor() {
28             super("The date range is not valid, make sure the from date is before the until date.");
29         }
30     }
31     
32     // Error for when the 'until' date is set in the future
33     class FutureUntilDateError extends InputError {

35             super("Future dates are not allowed for the until field");
