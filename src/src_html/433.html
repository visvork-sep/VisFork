1     import { useRef, useEffect, useState, memo } from "react";
2     import { schemeCategory10, select, } from "d3";
3     import cloud from "d3-cloud";
4     import { Word, processCommitMessages, lemmatizationFunction } from "./utils";
5     import { createTooltip } from "./Tooltip";
6     import { WordCloudData } from "@VisInterfaces/WordCloudData";
7     
8     
9     function WordCloud({ commitData }: WordCloudData) {
10     
11         const svgRef = useRef<SVGSVGElement>(null);
12         const [dimensions, setDimensions] = useState({ width: 800, height: 600 });
13         const [words, setWords] = useState<Word[]>([]);
14     
15         useEffect(() => {
16             const handleResize = () => {
17                 if (svgRef.current) {
18                     let { clientWidth, clientHeight } = svgRef.current;
19     
20                     // Set a maximum size to prevent it from being too lengthy
21                     const maxWidth = 800;
22                     const maxHeight = 600;
23     
24                     // Maintain aspect ratio
25                     if (clientWidth / clientHeight > maxWidth / maxHeight) {
26                         clientWidth = maxWidth;
27                         clientHeight = (maxWidth / clientWidth) * clientHeight;
28                     } else {
29                         clientHeight = maxHeight;
30                         clientWidth = (maxHeight / clientHeight) * clientWidth;
31                     }
32     
33                     setDimensions({ width: clientWidth, height: clientHeight });
34                 }
35             };
36     
37             window.addEventListener("resize", handleResize);
38             handleResize();
39     
40             return () => window.removeEventListener("resize", handleResize);
41         }, []);
42         useEffect(() => {
43             const newWords = processCommitMessages(commitData, lemmatizationFunction, 0, 20);
44             setWords(newWords);
45         }, [commitData]);
46     
47         useEffect(() => {
48             if (!svgRef.current) return;
49     
50             const layout = cloud<Word>()
51                 .size([dimensions.width, dimensions.height])
52                 .words(words)
53                 .padding(5)
54                 .rotate(() => (~~(Math.random() * 2) * 90))
55                 .font("Impact")
56                 .fontSize((d) => Math.sqrt(d.size) * 10)
57                 .on("end", draw);
58     
59             layout.start();
60     
61             function draw(words: Word[]) {
62                 const svg = select(svgRef.current);
63     
64                 // Clear previous word cloud
65                 svg.selectAll("*").remove();
66     
67                 const g = svg
68                     .attr("width", layout.size()[0])
69                     .attr("height", layout.size()[1])
70                     .append("g")
71                     .attr("transform", "translate(" + layout.size()[0] / 2 + "," + layout.size()[1] / 2 + ")");
72     
73                 const tooltip = createTooltip();
74     
75                 g.selectAll("text")
76                     .data(words)
77                     .enter().append("text")
78                     .style("font-size", (d) => d.size + "px")
79                     .style("font-family", "Impact")
80                     .style("fill", (_, i) => schemeCategory10[i % 10])
81                     .attr("text-anchor", "middle")
82                     .attr("transform", (d) => "translate(" + [d.x, d.y] + ")rotate(" + d.rotate + ")")
83                     .text((d) => d.text)
84                     .on("mouseover", (event, d) => {
85                         tooltip.transition().duration(200).style("opacity", 0.9);
86                         tooltip.html(
87                             `<strong>Word</strong>: ${d.text}<br>
88                             <strong>Frequency</strong>: ${d.frequency}`
89                         )
90                             .style("left", (event.pageX) + "px")
91                             .style("top", (event.pageY) + "px");
92                     })
93                     .on("mousemove", (event) => {
94                         tooltip.style("left", (event.pageX) + "px")
95                             .style("top", (event.pageY) + "px");
96                     })
97                     .on("mouseout", () => {
98                         tooltip.transition().duration(500).style("opacity", 0);
99                     });
100     
101                 return () => {
102                     tooltip.remove();
103                 };
104             }
105         }, [words, dimensions]);
106     
107         return <svg ref={svgRef} style={{ width: "100%", height: "100%" }}></svg>;
108     };
109     
110     export default memo(WordCloud);
