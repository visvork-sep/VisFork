19     /**
20      * Trim the input and remove forbidden characters from the input.
21      * 
22      * @param {string} str input
23      * @returns output - The sanitized string
24      * @returns conflicts - The forbidden characters found in the input
25      */
26     function sanitizeString(str: string): {output: string, conflicts: RegExpMatchArray | null} {
27         const pattern = /[^a-z0-9 /_.-]/gi;
28     
29         const output = str.replace(pattern, "").trim();
30         const conflicts = str.match(pattern);
31         return {
32             output,
33             conflicts
34         };
35     }

37     /**
38      * Validator for date strings.
39      * 
40      * rules: must be in the format "yyyy-mm-dd"
41      * 
42      * note: this function does not check for valid dates, only for the format.
43      * 
44      * @param input date string
45      * @returns true if the input is a valid date string 
46      */
47     function isValidDate(input: string): boolean {
48         const regex = /^\d{4}-(0[1-9]|1[012])-(0[1-9]|[12][0-9]|3[01])$/;
49     
50         return regex.test(input);
51     }

53     /**
54      * Validator for integer strings.
55      * 
56      * rules: no leading zeroes, no decimal points, no commas, no whitespace.
57      * 
58      * @param input integer string
59      * @returns true if the input is a valid integer string
60      */
61     function isValidInteger(input: string): boolean {
62         const regex = /^(0|[1-9][0-9]*)$/;
63     
64         return regex.test(input);
65     }

67     /**
68      * Prepare the repository input for passing to logic layer.
69      * 
70      * @param {String} input repository input
71      * @throws {ForbiddenCharactersError} if the input contains forbidden characters
72      * @throws {RepositorySyntaxError} if the input does not match the required syntax (Owner/RepositoryName)
73      * @returns owner and repository name
74      */
75     function prepareRepository(input: string): { owner: string, repositoryName: string } {
76         const { output, conflicts } = sanitizeString(input);
77         if (conflicts) {
78             throw new RepositoryInputErrors.ForbiddenCharactersError(conflicts);
79         }
80     
81         const words = output.split("/");
82     
83         if (words.length != 2) {
84             throw new RepositoryInputErrors.RepositorySyntaxError();
85         }
86     
87         const owner = words[0];
88         const repositoryName = words[1];
89     
90         return {
91             owner,
92             repositoryName
93         };
94     }

96     /**
97      * Prepare the forks count input for passing to logic layer.
98      * 
99      * @param {String} input forks count input
100      * @throws {NonIntegralError} if the input is not an integer
101      * @throws {ForbiddenCharactersError} if the input contains forbidden characters
102      * @throws {LessThanMinForksError} if the input is less than the minimum allowed forks
103      * @throws {GreaterThanMaxForksError} if the input is greater than the maximum allowed forks
104      * @returns forks count
105      */
106     function prepareForksCount(input?: string): number {
107         if (!input) {
108             throw new ForksCountInputErrors.NonIntegralError();
109         }
110     
111         const { output, conflicts } = sanitizeString(input);
112         if (conflicts) {
113             throw new ForksCountInputErrors.ForbiddenCharactersError(conflicts);
114         }
115     
116         if (!isValidInteger(output)) {
117             throw new ForksCountInputErrors.NonIntegralError();
118         }
119     
120         const num = Number(output);
121     
122         if (num < MIN_QUERIABLE_FORKS) {
123             throw new ForksCountInputErrors.LessThanMinForksError();
124         }
125     
126         if (num > MAX_QUERIABLE_FORKS) {
127             throw new ForksCountInputErrors.GreaterThanMaxForksError();
128         }
129     
130         return Number(output);
131     }

245      * Prepare the recently updated input for passing to logic layer.
246      *

248      * @throws {NonIntegralError} if the input is not an integer
249      * @throws {ForbiddenCharactersError} if the input contains forbidden characters
250      * @throws {OutOfRecentlyUpdatedRangeError} if the input is out of the allowed range
251      * @returns recently updated
252      */
253     function prepareRecentlyUpdated(input?: string): number | null {
254         if (!input) {
255             return null;
256         }
257     
258         const { output, conflicts } = sanitizeString(input);
259         if (conflicts) {
260             throw new RecentlyUpdatedInputErrors.ForbiddenCharactersError(conflicts);
261         }
262     
263         if (!isValidInteger(output)) {
264             throw new RecentlyUpdatedInputErrors.NonIntegralError();
265         }
266         
267         const num = Number(output);
268     
269         if (num < RECENT_ACTIVITY_MIN_MONTHS || num > RECENT_ACTIVITY_MAX_MONTHS) {
270             throw new RecentlyUpdatedInputErrors.OutOfRecentlyUpdatedRangeError();
271         }
