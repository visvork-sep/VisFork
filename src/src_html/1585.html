1     import { describe, it, expect, vi, afterEach } from "vitest";
2     import { render, screen, fireEvent, waitFor } from "@testing-library/react";
3     import { ThemeProvider } from "@primer/react";
4     import CommitTable from "./CommitTable";
5     import { CommitTableDetails } from "@VisInterfaces/CommitTableData";
6     
7     // Sample commit data
8     const mockCommits: CommitTableDetails[] = [
9         {
10             id: "abc1234def",
11             repo: "user1/repo1",
12             author: "Jane Doe",
13             login: "janedoe",
14             date: "2024-04-01",
15             message: "Initial commit"
16         },
17         {
18             id: "def5678ghi",
19             repo: "user2/repo2",
20             author: "John Smith",
21             login: "johnsmith",
22             date: "2024-04-02",
23             message: "Added feature"
24         },
25         {
26             id: "xyz9876uvw",
27             repo: "user3/repo3",
28             author: "Alice",
29             login: "aliceGH",
30             date: "2024-04-03",
31             message: "Fixed bug in authentication"
32         }
33     ];
34     
35     // Helper to render component with theme provider
36     const renderCommitTable = (data = mockCommits, handleSearchBarSubmission = vi.fn()) => {
37         return render(
38             <ThemeProvider>
39                 <CommitTable commitData={data} handleSearchBarSubmission={handleSearchBarSubmission} />
40             </ThemeProvider>
41         );
42     };
43     
44     afterEach(() => {
45         vi.restoreAllMocks();
46         document.body.innerHTML = "";
47     });
48     
49     describe("CommitTable", () => {
50         it("renders commit table with rows", () => {
51             renderCommitTable();
52     
53             expect(screen.getByText("Jane Doe")).toBeInTheDocument();
54             expect(screen.getByText("John Smith")).toBeInTheDocument();
55             expect(screen.getByText("Alice")).toBeInTheDocument();
56             expect(screen.getByText("Initial commit")).toBeInTheDocument();
57             expect(screen.getByText("Added feature")).toBeInTheDocument();
58         });
59     
60         it("filters commits based on search input", async () => {
61             renderCommitTable();
62     
63             const input = screen.getByLabelText("Search commit messages");
64             fireEvent.change(input, { target: { value: "feature" } });
65             const searchButton = screen.getByRole("button");
66             fireEvent.click(searchButton);
67     
68             await waitFor(() => {
69                 expect(screen.getByText("Added feature")).toBeInTheDocument();
70                 expect(screen.queryByText("Initial commit")).not.toBeInTheDocument();
71                 expect(screen.queryByText("Fixed bug in authentication")).not.toBeInTheDocument();
72             });
73         });
74     
75         it("sends filtered hashes to parent via handleSearchBarSubmission", async () => {
76             const mockHandler = vi.fn();
77             renderCommitTable(mockCommits, mockHandler);
78     
79             // should trigger once on initial mount
80             expect(mockHandler).toHaveBeenCalledWith([
81                 "abc1234def", "def5678ghi", "xyz9876uvw"
82             ]);
83     
84             const input = screen.getByLabelText("Search commit messages");
85             fireEvent.change(input, { target: { value: "auth" } });
86             const searchButton = screen.getByRole("button");
87             fireEvent.click(searchButton);
88     
89             // should call again with filtered result
90             await waitFor(() => {
91                 expect(mockHandler).toHaveBeenCalledWith(["xyz9876uvw"]);
92             });
93         });
94     
95         it("creates correct links for repo, author, and commit hash", () => {
96             renderCommitTable();
97     
98             const repoLink = screen.getByRole("link", { name: "user1/repo1" });
99             expect(repoLink).toHaveAttribute("href", "https://github.com/user1/repo1");
100     
101             const authorLink = screen.getByRole("link", { name: "Jane Doe" });
102             expect(authorLink).toHaveAttribute("href", "https://github.com/janedoe");
103     
104             const hashLink = screen.getByRole("link", { name: "abc1234" }); // short hash
105             expect(hashLink).toHaveAttribute("href", "https://github.com/user1/repo1/commit/abc1234def");
106         });
107     
108         it("renders nothing when filtered results are empty", async () => {
109             renderCommitTable();
110     
111             const input = screen.getByLabelText("Search commit messages");
112             fireEvent.change(input, { target: { value: "nonexistent term" } });
113             const searchButton = screen.getByRole("button");
114             fireEvent.click(searchButton);
115     
116             await waitFor(() => {
117                 const visibleMessages = screen.queryAllByText(/./); // grab any text node
118                 const content = visibleMessages.map(node => node.textContent).join(" ");
119                 expect(content).not.toMatch(/Jane Doe|John Smith|Alice/);
120             });
121         });
122     });
