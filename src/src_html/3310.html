21     function canCreateMainRepositoryInfo(
22         forkError: Error | null, 
23         commitError: Error | null, 
24         isLoadingFork: boolean, 
25         isLoadingCommits: boolean, 
26         simplifiedForkData: UnprocessedRepository[], 
27         filteredForks: UnprocessedRepository[],
28         simplifiedCommitData: UnprocessedCommitExtended[][], 
29         forkQueryState?: ForkQueryState
30     ): boolean {
31         return !forkError
32           && !commitError
33           && !isLoadingFork
34           && !isLoadingCommits
35           && simplifiedForkData.length > 0
36           && simplifiedCommitData.length === filteredForks.length
37           && !!forkQueryState?.owner
38           && !!forkQueryState?.repo;
39     }

41     function createMainRepositoryInfo(
42         filteredForks: UnprocessedRepository[], 
43         simplifiedCommitData: UnprocessedCommitExtended[][], 
44         forkQueryState: ForkQueryState
45     ): MainRepositoryInfo {
46         const completeForkData: RepositoryInfoWithCommits[] = filteredForks.map((fork, index) => ({
47             ...fork,
48             commits: simplifiedCommitData[index]
49         }));
50     
51         return {
52             owner: { login: forkQueryState.owner },
53             forks: completeForkData,
54             commits: [],
55             id: 0,
56             name: "",
57             description: null,
58             created_at: null,
59             last_pushed: null,
60             ownerType: "User",
61             defaultBranch: "main"
62         };
63     }

66     export function useFilteredData() {
67         // Create the state for the query parameters
68         const [forkQueryState, setForkQueryState] = useState<ForkQueryState | undefined>(undefined);
69     
70         // State for additional filtering, such as sorting and date range.
71         const [filters, setFilters] = useState<ForkFilter | undefined>(undefined);
72         const [finalForkData, setFinalForkData] = useState<UnprocessedRepository[]>([]);
73         const [finalCommitData, setFinalCommitData] = useState<UnprocessedCommitExtended[]>([]);
74     
75         const onRequestChange: FilterChangeHandler = (filters, forkQueryState) => {
76             setForkQueryState(forkQueryState);
77             setFilters(filters);
78         };
79     
80         const { data: forkData, isLoading: isLoadingFork, error: forkError } = useFetchForks(forkQueryState);
81     
82         const simplifiedForkData = useMemo(() => unwrap(forkData?.data).map(fork => toForkInfo(fork)), [forkData]);
83     
84     
85         const filteredForks = useMemo(() => {
86             return (filters) ? simplifiedForkData.filter(fork => isValidForkByFilter(fork, filters)) : [];
87         }, [simplifiedForkData, filters]);
88     
89     
90     
91         const commitResponses = useFetchCommitsBatch(filteredForks, forkQueryState);
92     
93         const { commitData, isLoading: isLoadingCommits, error: commitError } = commitResponses
94             .reduce<{ commitData?: GitHubAPICommit[][], isLoading: boolean, error: Error | null }>((acc, response) => {
95                 if ((acc.commitData && response.data?.data)) {
96                     acc.commitData.push(response.data.data);
97                 }
98     
99                 acc.isLoading = acc.isLoading || response.isLoading;
100     
101                 if (!acc.error) {
102                     acc.error = response.error;
103                 }
104     
105                 return acc;
106             },
107             {
108                 commitData: [],
109                 isLoading: false,
110                 error: null
111             });
112     
113         const simplifiedCommitData: UnprocessedCommitExtended[][] = useMemo(() => {
114             return unwrap(commitData).map(commits => commits.map(commit => toCommitInfo(commit)));
115         }, [commitData]);
116     
117         const mainRepositoryInfo = useMemo(() => {
118             const shouldCreate = canCreateMainRepositoryInfo(
119                 forkError, commitError, isLoadingFork, isLoadingCommits,
120                 simplifiedForkData, filteredForks, simplifiedCommitData, forkQueryState
121             );
122             
123             return (shouldCreate && forkQueryState)
124                 ? createMainRepositoryInfo(filteredForks, simplifiedCommitData, forkQueryState) : undefined;
125         }, [forkError, 
126             commitError, 
127             isLoadingFork, 
128             isLoadingCommits, 
129             simplifiedForkData, 
130             simplifiedCommitData, 
131             forkQueryState]);
132     
133         const flattenedCommits: UnprocessedCommitExtended[] =
134             mainRepositoryInfo ? mainRepositoryInfo.forks.reduce<UnprocessedCommitExtended[]>((acc, fork) => {
135                 const commits: UnprocessedCommitExtended[] = fork.commits.map(commit => ({
136                     ...commit,
137                     repo: `${fork.owner.login}/${fork.name}`,
138                     branch: fork.defaultBranch,
139     
140                 }));
141     
142                 acc = acc.concat(commits);
143                 return acc;
144             }, []) : [];
145     
146     
147         // Add final update only if all data has been loaded
148         useEffect(() => {
149             if (!isLoadingFork && !isLoadingCommits && filteredForks.length > 0 && flattenedCommits.length > 0) {
150                 setFinalForkData(filteredForks);
151                 setFinalCommitData(flattenedCommits);
152             }
153         }, [isLoadingFork, isLoadingCommits, filters]);
154     
155     
156     
157         return {
158             isLoading: isLoadingFork || isLoadingCommits,
159             forks: finalForkData,
160             commits: finalCommitData,
161             data: mainRepositoryInfo,
162             onFiltersChange: onRequestChange,
163             forkQuery: forkQueryState,
164             forkError,
165             commitError,
166             filters,
167             isLoadingFork,
168             isLoadingCommits
169         };
170     }
