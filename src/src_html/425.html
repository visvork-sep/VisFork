25     interface ParsedDataItem {
26         name: string;
27         type: string;
28         count: number;
29     };

31     //function which helpts define the nodeAlign values 
32     type AlignType = "justify" | "left" | "right" | "center";
33     
34     // Function to map string to d3-sankey alignment functions to define the nodeAlign value
35     const getNodeAlignFunction = (align: AlignType) => {
36         switch (align) {
37             case "left":
38                 return sankeyLeft;
39             case "right":
40                 return sankeyRight;
41             case "center":
42                 return sankeyCenter;
43             case "justify":
44             default:
45                 return sankeyJustify; // Default to justify
46         }
47     };

49     // Function to parse the data and return an array of ParsedDataItem
50     export function parseData({ commitData }: SankeyData): ParsedDataItem[] {
51         const parsedData: Record<string, ParsedDataItem> = {};
52     
53         // Loop through the data and create a key for each unique combination of repo and commit_type
54     
55         for (const dataPoint of commitData) {
56             const name = dataPoint.repo;
57             const type = dataPoint.commitType;
58             const key = `${name}-${type}`;
59     
60             // If the key already exists, increment the count, otherwise create a new entry
61             if (parsedData[key]) {
62                 parsedData[key].count++;
63             } else {
64                 parsedData[key] = { name, type, count: 1 };
65             }
66         }
67     
68         // Return the values of the parsedData object to be used as the data for the Sankey chart
69         return Object.values(parsedData);
70     }

72     // Main function to render the Sankey chart
73     export function SankeyChart(
74         {   // nodes and links defining the graph
75             links, // an iterable of link objects (typically [{source, target}, …])
76         }: { links: SankeyLinkMinimal<any, any>[]; },
77         {   // configuration options with defaults
78             width,
79             nodeGroup, // given n in nodes, returns an (ordinal) value for color
80         }: {
81             width: number;
82             nodeGroup?: (n: SankeyNodeMinimal<any, any>) => number;
83         }
84     ) {
85         let format = d3Format(","); // a function or format specifier for values in titles
86         const align = "justify"; // convenience shorthand for nodeAlign
87         let nodeGroups: Iterable<number> = []; // an array of ordinal values representing the node groups
88         const nodeAlign = getNodeAlignFunction(align); // Sankey node alignment strategy: left, right, justify, center
89         const nodeWidth = 15; // width of node rects
90         const nodePadding = 10; // vertical separation between adjacent nodes
91         const nodeLabel = undefined;
92         const nodeLabelPadding = 6; // horizontal separation between node and label
93         const nodeStroke = "currentColor"; // stroke around node rects
94         const nodeStrokeWidth = 1; // width of stroke around node rects, in pixels
95         const nodeStrokeOpacity = 1; // opacity of stroke around node rects
96         const nodeStrokeLinejoin = 1; // line join for stroke around node rects
97         const linkSource = 
98             (l: SankeyLinkMinimal<any, any>) => l.source; // given l in links, returns a node identifier string
99         const linkTarget = 
100             (l: SankeyLinkMinimal<any, any>) => l.target; // given l in links, returns a node identifier string
101         const linkValue = (l: SankeyLinkMinimal<any, any>) => l.value; // given l in links, returns the quantitative value
102         const linkPath = sankeyLinkHorizontal(); // given d in (computed) links, returns the SVG path
103         const linkTitle = (l: SankeyLinkMinimal<any, any>) =>
104             `${l.source.id} => ${l.target.id}\n${format(l.value)} commits`; // given d in (computed) links
105         let linkColor = "source-target"; // source, target, source-target, or static color
106         const linkStrokeOpacity = 0.5; // link stroke opacity
107         const linkMixBlendMode = "normal"; // link blending mode
108         const colors = schemeTableau10; // array of colors
109         const height = 400; // outer height, in pixels
110         const marginTop = 5; // top margin, in pixels
111         const marginRight = 1; // right margin, in pixels
112         const marginBottom = 5; // bottom margin, in pixels
113         const marginLeft = 1; // left margin, in pixels
114     
115     
116         //Mapping the soruce, target and value of the links
117         const LS = map(links, linkSource).map(intern);
118         const LT = map(links, linkTarget).map(intern);
119         const LV = map(links, linkValue);
120     
121         // If nodes is not defined, create a new array of nodes
122         const nodes = Array.from(union(LS, LT), (id, index) => ({ id, index, x0: 0, x1: 0, y0: 0, y1: 0 }));
123     
124         // Mapping the nodes to their id
125         const N = map(nodes as SankeyNode<{ id: string; }, any>[], (n) => n.id).map(intern);
126         const G = nodeGroup == null ? null : map(nodes, nodeGroup).map(intern);
127     
128         // Construct the links
129         links = map(links, (_, i) => ({
130             source: LS[i],
131             target: LT[i],
132             value: LV[i],
133         }));
134     
135         // Ignore a group-based linkColor option if no groups are specified.
136         if (!G && ["source", "target", "source-target"].includes(linkColor))
137             linkColor = "currentColor";
138     
139         // Compute default domains.
140         if (G && nodeGroups === undefined) nodeGroups = G;
141     
142         // Construct the scales.
143         const color = scaleOrdinal(nodeGroups, colors);
144     
145         // Computing the layout of the diagram using 
146     
147         sankey<{ id: string; }, any>()
148             .nodeId((d) => d.id)
149             .nodeAlign(nodeAlign)
150             .nodeWidth(nodeWidth)
151             .nodePadding(nodePadding)
152             .extent([
153                 [marginLeft, marginTop],
154                 [width - marginRight, height - marginBottom],
155             ])({ nodes: nodes as SankeyNode<{ id: string; }, any>[], links });
156     
157         // Compute titles and labels using layout nodes, so as to access aggregate values.
158         let Lt, Tl;
159         // eslint-disable-next-line prefer-const
160         ({ Lt, Tl, format } = computeTitlesAndLabels(format, nodeLabel, N, nodes, linkTitle, links));
161     
162         // A unique identifier for clip paths (to avoid conflicts).
163         const uid = `O-${Math.random().toString(16).slice(2)}`;
164     
165         // Constructing the SVG element.
166         const svg = create("svg")
167             .attr("width", width)
168             .attr("height", height)
169             .attr("viewBox", [0, 0, width, height])
170             .attr("style", "max-width: 100%; height: auto; height: intrinsic;");
171     
172         //Configure the original node looks 
173         const node = svg
174             .append("g")
175             .attr("stroke", nodeStroke)
176             .attr("stroke-width", nodeStrokeWidth)
177             .attr("stroke-opacity", nodeStrokeOpacity)
178             .attr("stroke-linejoin", nodeStrokeLinejoin)
179             .selectAll("rect")
180             .data(nodes)
181             .join("rect")
182             .attr("x", (d) => d.x0 ?? 0)
183             .attr("y", (d) => d.y0 ?? 0)
184             .attr("height", (d) => (d.y1 ?? 0) - (d.y0 ?? 0))
185             .attr("width", (d) => (d.x1 ?? 0) - (d.x0 ?? 0));
186     
187         //Coloring the nodes
188         if (G) node.attr("fill", ({ index: i }) => color(i !== undefined ? G[i] : undefined));
189     
190         //Configure the original look of the links 
191         const link = svg
192             .append("g")
193             .attr("fill", "none")
194             .attr("stroke-opacity", linkStrokeOpacity)
195             .selectAll("g")
196             .data(links)
197             .join("g")
198             .style("mix-blend-mode", linkMixBlendMode);
199     
200         //Coloring the links 
201         if (linkColor === "source-target")
202             link
203                 .append("linearGradient")
204                 .attr("id", (d) => `${uid}-link-${d.index}`)
205                 .attr("gradientUnits", "userSpaceOnUse")
206                 .attr("x1", (d) => d.source.x1)
207                 .attr("x2", (d) => d.target.x0)
208                 .call((gradient) =>
209                     gradient
210                         .append("stop")
211                         .attr("offset", "0%")
212                         .attr("stop-color", ({ source: { index: i } }) => color(G !== null ? G[i] : undefined))
213                 )
214                 .call((gradient) =>
215                     gradient
216                         .append("stop")
217                         .attr("offset", "100%")
218                         .attr("stop-color", ({ target: { index: i } }) => color(G !== null ? G[i] : undefined))
219                 );
220     
221         //Creating a gradient between the source and target nodes
222         createGradient(link, linkPath, linkColor, uid, color, G, Lt);
223     
224         //Setting up the label of the nodes
225         if (Tl)
226             svg
227                 .append("g")
228                 .attr("font-family", "sans-serif")
229                 .attr("font-size", 16)
230                 .attr("fill", "currentColor")
231                 .selectAll("text")
232                 .data(nodes)
233                 .join("text")
234                 .attr("x", (d) =>
235                     (d.x0 ?? 0) < width / 2 ? (d.x1 ?? 0) + nodeLabelPadding : (d.x0 ?? 0) - nodeLabelPadding
236                 )
237                 .attr("y", (d) => ((d.y1 ?? 0) + (d.y0 ?? 0)) / 2)//THE 0 IS ADDED JUST TO NOT HAVE UNDEFINED. CHANGE LATER
238                 .attr("dy", "0.35em")
239                 .attr("text-anchor", (d) => ((d.x0 ?? 0) < width / 2 ? "start" : "end"))//THE 0 IS ADDED 
240                 // JUST TO NOT HAVE UNDEFINED. CHANGE LATER
241                 .text(({ index: i }) => i !== undefined ? Tl[i] : null);
242     
243         //Setting up the hovering effect for the nodes 
244         node
245             .on("mouseover", function () {
246                 select(this).attr("stroke", "black");
247                 select(this).attr("stroke-width", 3);
248                 select(this).attr("stroke-opacity", 1);
249                 select(this).attr("stroke-linejoin", "round");
250             })
251             .on("mouseout", function () {
252                 select(this).attr("stroke", nodeStroke);
253                 select(this).attr("stroke-width", nodeStrokeWidth);
254                 select(this).attr("stroke-opacity", nodeStrokeOpacity);
255                 select(this).attr("stroke-linejoin", nodeStrokeLinejoin);
256             });
257     
258         //Setting up the hovering efect for the links 
259         link
260             .on("mouseover", function () {
261                 select(this).attr("stroke", "black");
262                 select(this).attr("stroke-opacity", 1);
263             })
264             .on("mouseout", function () {
265                 select(this).attr("stroke", linkColor);
266                 select(this).attr("stroke-opacity", linkStrokeOpacity);
267             });
268     
269         //Function to convert the value to a string
270         function intern(val: any) {
271             return val !== null && typeof val === "object"
272                 ? val.valueOf()
273                 : val;
274         }
275     
276         //Return the SVG element
277         const svgNode = svg.node();
278         if (svgNode) {
279             return Object.assign(svgNode, { scales: { color } });
280         }
281         return null;
282     }

284     function computeTitlesAndLabels(format: (n: number | { valueOf(): number; }) => string, 
285         nodeLabel: undefined, N: any[], 
286         nodes: { id: any; index: number; x0: number; x1: number; y0: number; y1: number; }[], 
287         linkTitle: (l: SankeyLinkMinimal<any, any>) => string, 
288         links: SankeyLinkMinimal<any, any>[]) {
289         if (typeof format !== "function") format = d3Format(format);
290         const Tl = nodeLabel === undefined
291             ? N
292             : nodeLabel == null
293                 ? null
294                 : map(nodes, nodeLabel);
295         const Lt = linkTitle == null ? null : map(links, linkTitle);
296         return { Lt, Tl, format };
297     }

299     function createGradient(link: any, 
300         linkPath: any, 
301         linkColor: string, 
302         uid: string, 
303         color: any, 
304         G: any[] | null, 
305         Lt: string[] | null) {
306         link
307             .append("path")
308             .attr("d", linkPath)
309             .attr(
310                 "stroke",
311                 linkColor === "source-target"
312                     ? ({ index: i }: { index: number}) => `url(#${uid}-link-${i})`
313                     : linkColor === "source"
314                         ? ({ source: { index: i }}: { source: { index: number } }) => color(G !== null ? G[i] : undefined)
315                         : linkColor === "target"
316                             ? ({ target: { index: i } }: { target: { index: number } }) => 
317                                 color(G !== null ? G[i] : undefined)
318                             : linkColor
319             )
320             .attr("stroke-width", ({ width }: { width: number }) => Math.max(1, width ?? 640))
321             .call(
322                 Lt
323                     ? (path: d3.Selection<SVGPathElement, any, any, any>) => 
324                         path.append("title").text(({ index: i }: { index: number }) => 
325                             i !== undefined ? Lt[i] : null)
326                     // eslint-disable-next-line @typescript-eslint/no-empty-function
327                     : () => { }
328             );
329     }

331     // SankeyChartBuild component
332     export function SankeyDiagram(data: SankeyData) {
333         useEffect(() => {
334     
335             if (!data) {
336                 return;
337             }
338     
339             const container = document.getElementById("sankey-diagram");
340             if (!container) {
341                 return;
342             }
343     
344             const width = container.clientWidth;
345     
346             // show user when no data is selected
347             if (!data?.commitData?.length) {
348                 select("#sankey-diagram").selectAll("*").remove();
349                 select("#sankey-diagram")
350                     .append("text")
351                     .attr("x", 200)
352                     .attr("y", 200)
353                     .attr("fill", "black")
354                     .attr("text-anchor", "middle")
355                     .attr("font-size", "16px")
356                     .text("No data selected");
357                 return;
358             }
359     
360             // Parse the data and create the Sankey chart
361             const sankeyData = parseData(data);
362             const links = sankeyData.map((d) => ({
363                 source: d.name,
364                 target: d.type,
365                 value: d.count,
366             }));
367     
368             // Create the Sankey chart
369             const chart = SankeyChart(
370                 { links },
371                 {
372                     width,
373                     nodeGroup: (d) => parseInt((d.index ? d.index.toString() : "").split(/\W/)[0]),
374                 }
375             );
376     
377             // Clear previous chart and append new one
378             select("#sankey-diagram").selectAll("*").remove();
379             select("#sankey-diagram").append(() => chart);
380         }, [data]);
381     
382         return <div id="sankey-diagram" data-testid="sankey-diagram" style={{ width: "100%", height: "400px" }}></div>;
383     };
