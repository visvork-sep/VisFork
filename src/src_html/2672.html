23     export function drawLanes(
24         g: Selection<SVGGElement, unknown, null, undefined>,
25         lanes: Record<string, { minY: number; maxY: number }>,
26         svgWidth: number,
27         isDarkMode: boolean,
28         darkColor: string,
29         lightColor: string
30     ) {
31         const backgrounds = g.append("g").attr("class", "repo-backgrounds");
32       
33         Object.entries(lanes).forEach(([repo, { minY: minX, maxY: maxX }], i) => {
34             const laneColor = i % 2 === 0 ? darkColor : lightColor;
35       
36             backgrounds.append("rect")
37                 .attr("x", -c.MARGIN.left)
38                 .attr("y", minX - c.NODE_RADIUS)
39                 .attr("width", svgWidth)
40                 .attr("height", maxX - minX + c.NODE_RADIUS * 2)
41                 .attr("fill", laneColor)
42                 .attr("opacity", 0.4);
43       
44             backgrounds.append("text")
45                 .attr("x", -c.MARGIN.left + 5)
46                 .attr("y", (minX - c.NODE_RADIUS) + (maxX - minX + c.NODE_RADIUS * 2) / 2)
47                 .attr("text-anchor", "start")
48                 .attr("alignment-baseline", "middle")
49                 .text(repo.split("/")[0])
50                 .attr("fill", `${isDarkMode ? "white" : "black"}`);
51         });
52     }

54     export function drawTimelineMarkers(
55         g: Selection<SVGGElement, unknown, null, undefined>,
56         sortedNodes: GraphNode<Commit | GroupedNode, unknown>[],
57         totalHeight: number,
58         isDarkMode: boolean,
59         colorMarker: string,
60         merged: boolean
61     ) {
62         if (!merged) {
63             const formatMonth = timeFormat("%b");
64             const formatYear = timeFormat("%Y");
65             let lastMonth = "";
66             let lastYear = "";
67             
68             const monthGroup = g.append("g").attr("class", "month-lines");
69             let lastLabelX = -Infinity;
70             
71             for (const node of sortedNodes) {
72                 const currentDate = new Date(node.data.date);
73                 const currentMonth = formatMonth(currentDate);
74                 const currentYear = formatYear(currentDate);
75                 const labelX = node.x;
76             
77                 if (currentMonth !== lastMonth) {
78                     // always draw the vertical line
79                     monthGroup.append("line")
80                         .attr("x1", labelX)
81                         .attr("x2", labelX)
82                         .attr("y1", 0)
83                         .attr("y2", totalHeight - 10)
84                         .attr("stroke", colorMarker)
85                         .attr("stroke-dasharray", "3,3");
86             
87                     // only add text label if we have enough space
88                     if (Math.abs(labelX - lastLabelX) > c.MIN_LABEL_SPACING) {
89                         const isNewYear = currentYear !== lastYear;
90                         const labelText = isNewYear 
91                             ? `${currentMonth} ${currentYear}`
92                             : currentMonth;
93             
94                         monthGroup.append("text")
95                             .attr("x", labelX)
96                             .attr("y", totalHeight + c.MARGIN.bottom - 5)
97                             .attr("font-size", 12)
98                             .style("text-anchor", "middle")
99                             .style("fill", `${isDarkMode? "white" : "black"}`)
100                             .text(labelText);
101             
102                         lastLabelX = labelX;
103                     }
104             
105                     lastMonth = currentMonth;
106                     lastYear = currentYear;
107                 }
108             }
109         }
110     }

112     export function drawEdgeCurve(d: MutGraphLink<Commit | GroupedNode, undefined>) {
113         if (d.source.y < d.target.y) {
114             return `
115                     M${d.source.x},${d.source.y}
116                     L${d.source.x},${d.target.y - c.CURVE_SIZE}
117                     C${d.source.x},${d.target.y}
118                     ${d.source.x},${d.target.y}
119                     ${d.source.x + c.CURVE_SIZE},${d.target.y}
120                     L${d.target.x},${d.target.y}
121                 `;
122         } else if (d.source.y === d.target.y) {
123             return `
124                     M${d.source.x},${d.source.y}
125                     L${d.target.x},${d.target.y}
126                 `;
127         } else {
128             return `
129                     M${d.source.x},${d.source.y}
130                     L${d.target.x - c.CURVE_SIZE},${d.source.y}
131                     C${d.target.x},${d.source.y}
132                     ${d.target.x},${d.source.y}
133                     ${d.target.x},${d.source.y - c.CURVE_SIZE}
134                     L${d.target.x},${d.target.y}
135                 `;
136         }
137     }

139     export function drawMergedNodes(
140         g: Selection<SVGGElement, unknown, null, undefined>,
141         colorMap: Map<string, string>,
142         mergedNodes: MutGraphNode<GroupedNode, undefined>[]) {
143         
144         const mergedCircles = mergedNodes.filter(node => node.data.branch === "forkParent");
145         const circles = g.append("g")
146             .selectAll("circle")
147             .data(mergedCircles)
148             .enter()
149             .append("circle")
150             .attr("cx", d => d.x ?? 0)
151             .attr("cy", d => d.y ?? 0)
152             .attr("r", c.NODE_RADIUS)
153             .style("cursor", "pointer")
154             .attr("fill", d => colorMap.get(d.data.repo) ?? "999");
155     
156         const mergedSquares = mergedNodes.filter(node => node.data.branch === "default");
157         const squares = g.append("g")
158             .selectAll("rect")
159             .data(mergedSquares)
160             .enter()
161             .append("rect")
162             .attr("x", d => (d.x ?? 0) - c.NODE_RADIUS)
163             .attr("y", d => (d.y ?? 0) - c.NODE_RADIUS)
164             .attr("width", c.NODE_RADIUS * 2)
165             .attr("height", c.NODE_RADIUS * 2)
166             .attr("fill", d => colorMap.get(d.data.repo) ?? "999");
167     
168         const mergedTriangles = mergedNodes.filter(node => node.data.branch === "merge");
169         const triangles = g.append("g")
170             .selectAll("polygon")
171             .data(mergedTriangles)
172             .enter()
173             .append("polygon")
174             .attr("points", `0,-${c.NODE_RADIUS} ${c.NODE_RADIUS},${c.NODE_RADIUS} -${c.NODE_RADIUS},${c.NODE_RADIUS}`)
175             .attr("transform", d => {
176                 const x = d.x ?? 0;
177                 const y = d.y ?? 0;
178                 return `translate(${x},${y})`;
179             })
180             .style("cursor", "pointer")
181             .attr("fill", d => colorMap.get(d.data.repo) ?? "999");
182     
183         return {circles, squares, triangles};
184     }

206     export function drawLegends( 
207         merged : boolean, 
208         legend: Selection<BaseType, unknown, HTMLElement, undefined>, 
209         colorMap: Map<string, string>,
210         shapeColor: string,
211         sortedNodes: MutGraphNode<Commit | GroupedNode, undefined>[],
212         setSelectAll: React.Dispatch<React.SetStateAction<boolean>>,
213         handle: (commitIds: string[]) => void,
214         resetBrushing: () => void) {
215             
216         const colorLegend = legend.append("div").attr("id", "color-legend");
217     
218         colorMap.forEach((colorValue, repoName) => {
219             const div = colorLegend
220                 .append("div")
221                 .style("display", "flex")
222                 .style("align-items", "center")
223                 .style("margin-right", "8px"); // small spacing between color items
224     
225             div
226                 .append("svg")
227                 .style("overflow", "visible")
228                 .attr("width", c.LEGEND_SIZE)
229                 .attr("height", c.LEGEND_SIZE)
230                 .append("circle")
231                 .attr("cx", c.LEGEND_SIZE / 2)
232                 .attr("cy", c.LEGEND_SIZE / 2)
233                 .attr("r", c.LEGEND_SIZE / 2)
234                 .style("cursor", "pointer")
235                 .on("click", function() {
236                     const selected = sortedNodes.filter(node => node.data.repo === repoName)
237                         .flatMap((node) =>
238                             merged
239                                 ? (node as MutGraphNode<GroupedNode, unknown>).data.nodes
240                                 : [node.data.id]
241                         );
242                     setSelectAll(false);
243                     resetBrushing();
244                     handle(selected);
245                 }) 
246                 .attr("fill", colorValue);
247     
248             div
249                 .append("text")
250                 .text(repoName)
251                 .style("margin-left", c.LEGEND_TEXT_MARGIN);
252         });
253     
254         if (merged) {
255             const shapeLegend = legend
256                 .append("div")
257                 .attr("id", "shape-legend")
258                 .style("margin-left", c.LEGENDS_SPACING); // spacing to the right of color legend
259     
260             const shapeLegendData = [
261                 { label: "Fork/Merge parent", shape: symbolCircle },
262                 { label: "Commit(s) without deviations", shape: symbolSquare },
263                 { label: "Merge commit" , shape: symbolTriangle },
264             ];
265     
266             shapeLegendData.forEach(({ label, shape }) => {
267                 const item = shapeLegend
268                     .append("div")
269                     .style("display", "flex")
270                     .style("align-items", "center")
271                     .style("margin-right", "8px")
272                     .style("margin-bottom", "4px");
273     
274                 item
275                     .append("svg")
276                     .attr("width", c.LEGEND_SIZE)
277                     .attr("height", c.LEGEND_SIZE)
278                     .style("flex-shrink", "0")
279                     .style("cursor", "pointer")
280                     .append("path")
281                     .attr("transform", `translate(${c.LEGEND_SIZE / 2}, ${c.LEGEND_SIZE / 2})`)
282                     .attr("d",symbol().type(shape).size(c.LEGEND_SYMBOL_SIZE))
283                     .on("click", function() {
284                         const selected = (sortedNodes as MutGraphNode<GroupedNode, undefined>[])
285                             .filter(node => node.data.branch === 
286                                 (label === "Fork/Merge parent" ? "forkParent" :
287                                     label === "Merge commit" ? "merge" : "default"
288                                 )
289                             )
290                             .flatMap(node => node.data.nodes);
291                         setSelectAll(false);
292                         resetBrushing();
293                         handle(selected);
294                     }) 
295                     .attr("fill", shapeColor);
296     
297                 item
298                     .append("text")
299                     .text(label)
300                     .style("margin-left", c.LEGEND_TEXT_MARGIN);
301             });
302         }
