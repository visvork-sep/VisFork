4     export function toForkInfo(fork: GitHubAPIFork): UnprocessedRepository {
5         return {
6             id: fork.id,
7             name: fork.name,
8             owner: { login: fork.owner.login },
9             description: fork.description,
10             created_at: fork.created_at ? new Date(fork.created_at) : null,
11             last_pushed: fork.pushed_at ? new Date(fork.pushed_at) : null,
12             ownerType: fork.owner.type === "Organization" ? "Organization" : "User",

14         };
