44                     [fork].filter((fork => isValidForkByFilter(fork, filter)))
45                 ).toStrictEqual([fork]);
46             });
47     
48             it("should return an array containing no forks - ownerType is not valid", () => {
49                 filter.ownerTypes = ["User"];
50                 fork.ownerType = "Organization";
51                 expect(
52                     [fork].filter((fork => isValidForkByFilter(fork, filter)))
53                 ).toStrictEqual([]);
54             });
55         });
56     
57         describe("Updated in last n months", () => {
58             it("should return an array containing the same fork(s) - nrOfMonths is undefined", () => {
59                 filter.updatedInLastMonths = undefined;
60                 expect(
61                     [fork].filter((fork => isValidForkByFilter(fork, filter)))
62                 ).toStrictEqual([fork]);
63             });
64     
65             it("should return empty array - fork last update was NOT within last n months", () => {
66                 filter.updatedInLastMonths = 1; // 20 years
67                 fork.last_pushed = new Date("2022-09-14T06:51:16Z");

69                     [fork].filter((fork => isValidForkByFilter(fork, filter)))

72     
73             it("should return an array containing the same fork(s) - fork last update was within last n months", () => {
