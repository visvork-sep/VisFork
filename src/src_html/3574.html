1     import { useState, useCallback } from "react";
2     import { FilterFormState } from "../Types/UIFormTypes";
3     import {
4         COMMIT_TYPES,
5         FORKS_COUNT_INPUT_INITIAL,
6         FORKS_SORTING_ORDERS,
7         OWNER_TYPES,
8         SORT_DIRECTION,
9         ForksSortingOrder,
10         SortDirection,
11         CommitType,
12         OwnerType
13     }
14         from "@Utils/Constants";
15     import { assert } from "@Utils/Assert";
16     
17     // Initial state for the filter form
18     const initialForm: FilterFormState = {
19         repository: "", // Repository name input
20         forksCount: FORKS_COUNT_INPUT_INITIAL, // Default fork count input
21         forksOrder: FORKS_SORTING_ORDERS.STARGAZERS.value, // Default sorting order (by stargazers)
22         forksAscDesc: SORT_DIRECTION.ASCENDING.value, // Default sorting direction (ascending)
23         commitTypeFilter: Object.values(COMMIT_TYPES).map(t => t.value), // Default commit type filter (all types selected)
24         ownerTypeFilter: Object.values(OWNER_TYPES).map(t => t.value), // Default owner type filter (all types selected)
25         commitsDateRangeFrom: (() => {
26             const date = new Date();
27             date.setFullYear(date.getFullYear() - 1);
28             return date.toISOString().split("T")[0];
29         })(), // Start date for commits filter
30         commitsDateRangeUntil: new Date().toISOString().split("T")[0], // End date for commits filter
31         recentlyUpdated: "" // Recently updated filter input
32     };
33     
34     function useFilterForm() {
35         const [form, setForm] = useState<FilterFormState>(initialForm);
36     
37         /**
38          * Updates the repository input field in the form state.
39          * @param {string} input - The new repository name entered by the user.
40          */
41         const handleRepositoryChange = useCallback((input: string) => {
42             setForm((prev) => ({
43                 ...prev,
44                 repository: input
45             }));
46         }, []);
47     
48         /**
49          * Updates the forks count filter in the form state.
50          * @param {string} input - The new fork count input as a string.
51          */
52         const handleForksCountChange = useCallback((input: string) => {
53             setForm((prev) => ({ ...prev, forksCount: input }));
54         }, []);
55     
56         /**
57          * Updates the forks sorting order.
58          * @param {string} value - 
59          * The selected sorting order. Must be one of the defined sorting options in FORKS_SORTING_ORDERS.
60          */
61         const handleForksOrderChange = useCallback((value: string) => {
62             assert(
63                 value === FORKS_SORTING_ORDERS.WATCHERS.value ||
64                 value === FORKS_SORTING_ORDERS.STARGAZERS.value ||
65                 value === FORKS_SORTING_ORDERS.NEWEST.value ||
66                 value === FORKS_SORTING_ORDERS.OLDEST.value,
67                 "Developer error: Invalid sorting order selected");
68     
69             setForm((prev) => ({ ...prev, forksOrder: value as ForksSortingOrder }));
70         }, []);
71     
72         /**
73          * Updates the sorting direction (ascending/descending) for forks.
74          * @param {string} value - 
75          * The selected sorting direction. Must be one of the defined sorting directions in SORT_DIRECTION.
76          */
77         const handleForksOrderAscDescChange = useCallback((value: string) => {
78             assert(value === SORT_DIRECTION.ASCENDING.value || value === SORT_DIRECTION.DESCENDING.value,
79                 "Developer error: Invalid sorting direction selected");
80             setForm((prev) => ({ ...prev, forksAscDesc: value as SortDirection }));
81         }, []);
82     
83         /**
84          * Updates the start date for commit filtering.
85          * @param {string} input - The new start date entered by the user.
86          */
87         const handleCommitsDateRangeFromChange = useCallback((input: string) => {
88             setForm((prev) => ({ ...prev, commitsDateRangeFrom: input }));
89         }, []);
90     
91         /**
92          * Updates the end date for commit filtering.
93          * @param {string} input - The new end date entered by the user.
94          */
95         const handleCommitsDateRangeUntilChange = useCallback((input: string) => {
96             setForm((prev) => ({ ...prev, commitsDateRangeUntil: input }));
97         }, []);
98     
99         /**
100          * Updates the selected commit types for filtering.
101          * 
102          * @param {string[]} selected - 
103          * An array of selected commit type values. Each value must be present in the predefined COMMIT_TYPES.
104          */
105         const handleCommitsTypeFilterChange = useCallback((selected: string[]) => {
106             assert(selected.every(s =>
107                 (s === COMMIT_TYPES.ADAPTIVE.value ||
108                 s === COMMIT_TYPES.CORRECTIVE.value ||
109                 s === COMMIT_TYPES.PERFECTIVE.value ||
110                 s === COMMIT_TYPES.UNKNOWN.value)
111     
112             ), "Developer error: Invalid commit type selected");
113     
114             setForm((prev) => ({ ...prev, commitTypeFilter: selected as CommitType[] }));
115         }, []);
116     
117         /**
118          * Updates the selected owner types for filtering.
119          * 
120          * @param {string[]} selected - 
121          * An array of selected owner type values. Each value must be present in the predefined OWNER_TYPES.
122          */
123         const handleOwnerTypeFilterChange = useCallback((selected: string[]) => {
124             assert(selected.every(s =>
125                 (s === OWNER_TYPES.USER.value ||
126                 s === OWNER_TYPES.ORGANIZATION.value)
127             ), "Developer error: Invalid owner type selected");
128             setForm((prev) => ({ ...prev, ownerTypeFilter: selected as OwnerType[] }));
129         }, []);
130     
131         /**
132          * Updates the recently updated filter input.
133          * @param {string} input - The new value for the recently updated filter.
134          */
135         const handleRecentlyUpdatedChange = useCallback((input: string) => {
136             setForm((prev) => ({ ...prev, recentlyUpdated: input }));
137         }, []);
138     
139         return {
140             form,
141             handleRepositoryChange,
142             handleForksCountChange,
143             handleForksOrderChange,
144             handleForksOrderAscDescChange,
145             handleCommitsDateRangeFromChange,
146             handleCommitsDateRangeUntilChange,
147             handleCommitsTypeFilterChange,
148             handleOwnerTypeFilterChange,
149             handleRecentlyUpdatedChange,
150         };
151     }
152     
153     export {
154         useFilterForm
155     };
