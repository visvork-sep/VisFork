1     import { Column, DataTable, Table } from "@primer/react/experimental";
2     import { Box, Link, useTheme } from "@primer/react";
3     import { useState, useMemo, memo, useEffect } from "react";
4     import { CommitTableDetails, CommitTableProps } from "@VisInterfaces/CommitTableData";
5     import SearchBar from "./SearchBar";
6     
7     function CommitTable({ commitData, handleSearchBarSubmission }: CommitTableProps) {
8         // Fetch current color mode (light or dark)
9         const { colorMode } = useTheme();
10     
11     
12         // Memoize hyperlink color based on color mode
13         const linkColor = useMemo(() => {
14             return colorMode === "dark" || colorMode === "night" ? "white" : "black";
15         }, [colorMode]);
16     
17         const [searchTerm, setSearchTerm] = useState("");
18     
19         const onSearch = (term: string) => {
20             setSearchTerm(term); // Update the search term state
21         };
22     
23         const preprocessedData = useMemo(() => {
24             return commitData.map(commit => ({
25                 ...commit,
26                 lowerMessage: commit.message.toLowerCase() // Store lowercase message
27             }));
28         }, [commitData]);
29     
30         // Memoize filtered data to prevent unnecessary recalculations
31         const filteredData = useMemo(() => {
32             const lowerSearch = searchTerm.toLowerCase();
33             return preprocessedData.filter(commit =>
34                 commit.lowerMessage.includes(lowerSearch)
35             );
36         }, [preprocessedData, searchTerm]);
37     
38         useEffect(() => {
39             // Extract hashes from filtered data
40             const hashes = filteredData.map(commit => commit.id);
41             handleSearchBarSubmission(hashes); // Pass the hashes to the parent component
42         }, [filteredData, handleSearchBarSubmission]);
43     
44         // Memoize column definitions
45         const columns: Column<CommitTableDetails>[] = useMemo(() => [
46             {
47                 header: "Owner/Repo",
48                 field: "repo",
49                 rowHeader: true,
50                 width: "auto",
51                 renderCell: (row: CommitTableDetails) => (
52                     <Link
53                         href={`https://github.com/${row.repo}`}
54                         target="_blank"
55                         rel="noopener noreferrer"
56                         sx={{ color: linkColor, textDecoration: "none" }}
57                     >
58                         {row.repo}
59                     </Link>
60                 ),
61             },
62             {
63                 header: "Author",
64                 field: "author",
65                 rowHeader: true,
66                 width: "auto",
67                 renderCell: (row: CommitTableDetails) => (
68                     <Link
69                         // This is the link to the author's GitHub profile using the username
70                         href={`https://github.com/${row.login}`}
71                         target="_blank"
72                         rel="noopener noreferrer"
73                         sx={{ color: linkColor, textDecoration: "none" }}
74                     >
75                         {row.author}
76                     </Link>
77                 ),
78             },
79             {
80                 header: "Commit Date",
81                 field: "date",
82                 rowHeader: true,
83                 width: "auto",
84             },
85             {
86                 header: "Commit Message",
87                 field: "message",
88                 rowHeader: true,
89                 width: "auto",
90             },
91             {
92                 header: "Hash",
93                 field: "id",
94                 rowHeader: true,
95                 width: "auto",
96                 renderCell: (row: CommitTableDetails) => (
97                     <Box
98                         sx={{
99                             whiteSpace: "normal",
100                             wordBreak: "break-all",
101                             overflowWrap: "break-word",
102                             minWidth: "50px",
103                         }}
104                     >
105                         <Link
106                             href={`https://github.com/${row.repo}/commit/${row.id}`}
107                             target="_blank"
108                             rel="noopener noreferrer"
109                             sx={{ color: linkColor, textDecoration: "none" }}
110                         >
111                             {row.id.substring(0, 7)}
112                         </Link>
113                     </Box>
114                 ),
115             },
116         ], [linkColor]);
117     
118         return (
119             <Box>
120                 <SearchBar onSearch={onSearch} />
121     
122                 <Box sx={{ maxHeight: "510px", overflowY: "auto" }}>
123                     <Table.Container>
124                         <DataTable data={filteredData} columns={columns} />
125                     </Table.Container>
126                 </Box>
127             </Box>
128         );
129     };
130     
131     export default memo(CommitTable);
