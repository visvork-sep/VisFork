20     * Fetch the amount of commits in a certain repository
21     */
22     export async function fetchCommitCount(parameters: CommitQueryParams, accessToken: string, page = 1) {
23     
24         const response = await fetchClient.GET("/repos/{owner}/{repo}/commits", {
25             params: {
26                 ...parameters,
27                 query: {
28                     ...parameters.query, // Keep existing query params
29                     per_page: PAGE_SIZE,
30                     page
31                 }
32             },
33             headers: {
34                 Authorization: `Bearer ${accessToken}`,
35             },
36         });
37     
38         const linkHeader = response.response.headers.get("link");
39         if (linkHeader) {
40             const lastPageMatch = linkHeader.match(/&page=(\d+)>; rel="last"/);
41             if (lastPageMatch) {
42                 const lastPage = parseInt(lastPageMatch[1]);
43                 // Each page contains a hundred commits
44                 return lastPage * PAGE_SIZE;
45             }
46         }
47     
48         // If no link header is found, it means there is at most one page of results
49         // and we can return the PAGE_SIZE as the maximum number of commits
50         return PAGE_SIZE;
51     }

53     export async function fetchCommits(parameters: CommitQueryParams, accessToken: string, page = 1) {
54         const allCommits: components["schemas"]["commit"][] = [];
55         let response;
56         let pagesRemaining = true;
57     
58         while (pagesRemaining) {
59             response = await fetchClient.GET("/repos/{owner}/{repo}/commits", {
60                 params: {
61                     ...parameters,
62                     query: {
63                         ...parameters.query, // Keep existing query params
64                         per_page: PAGE_SIZE,
65                         page
66                     }
67                 },
68                 headers: {
69                     Authorization: `Bearer ${accessToken}`,
70                 },
71             });
72     
73     
74             if (response?.data) {
75                 allCommits.push(...response.data); // Collect all commits in a single array:q
76             }
77             const linkHeader = response.response.headers.get("link");
78     
79             pagesRemaining = linkHeader?.includes("rel=\"next\"") ?? false;
80             page++;
81     
82         }
83     
84         // Return a single FetchResponse-like object with all the data
85         return {
86             ...response, // Retain original FetchResponse structure
87             data: allCommits, // Replace data with the accumulated commits
88         };
89     }

91     async function fetchForksPage(parameters: ForkQueryParams, 
92         accessToken: string, 
93         forksNumber: number, 
94         allForks: GitHubAPIFork[], 
95         page: number) {
96         let pagesRemaining = true;
97         let response;
98         
99         while (pagesRemaining && allForks.length < forksNumber) {
100             response = await fetchClient.GET("/repos/{owner}/{repo}/forks", {
101                 params: {
102                     ...parameters,
103                     query: {
104                         ...parameters.query, // Keep existing query params
105                         per_page: PAGE_SIZE, // Prevent exceeding limit
106                         page
107                     }
108                 },
109                 headers: {
110                     Authorization: `Bearer ${accessToken}`,
111                 },
112             });
113     
114             if (response?.data) {
115                 allForks.push(...response.data);
116             }
117     
118             if (allForks.length >= forksNumber) break;
119     
120             const linkHeader = response.response.headers.get("link");
121             pagesRemaining = linkHeader?.includes("rel=\"next\"") ?? false;
122             page++;
123         }
124         
125         return { response, allForks };
126     }

128     /**
129      * Fetch forks using REST API.
130      */
131     export async function fetchForks(parameters: ForkQueryParams, accessToken: string, forksNumber = 5) {
132         const allForks: GitHubAPIFork[] = [];
133     
134         // Fetch main repository, not just the forks
135         const mainRepoResponse = await fetchClient.GET("/repos/{owner}/{repo}", {
136             params: {
137                 // Parameters for main repo only need the repo and ownder info
138                 path: parameters.path,
139             },
140             headers: {
141                 Authorization: `Bearer ${accessToken}`,
142             },
143         });
144     
145         if (mainRepoResponse?.data) {
146             allForks.push(mainRepoResponse.data as GitHubAPIFork);
147         }
148     
149         // Fetch all fork repositories
150         const { response } = await fetchForksPage(parameters, accessToken, forksNumber, allForks, 1);
151     
152         // Return a combined response
153         return {
154             ...response, // Retain original response structure
155             data: allForks.slice(0, forksNumber + 1), // Replace data with accumulated main repo + forks
156         };
157     }
