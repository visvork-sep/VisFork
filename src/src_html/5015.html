14     function safePrepare<T, U>(
15         prepareFunction: (input: T) => U,
16         input: T,
17         errorSetter: Dispatch<SetStateAction<InputError | null>>
18     ) {
19         try {
20             const output = prepareFunction(input);
21             errorSetter(null);
22             return output;
23         } catch (e) {
24             setInputError(e, errorSetter);
25             return null;
26         }
27     }
